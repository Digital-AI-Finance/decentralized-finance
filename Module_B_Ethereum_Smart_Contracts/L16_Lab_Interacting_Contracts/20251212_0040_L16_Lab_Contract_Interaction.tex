\begin{frame}{Exercise 1: Deploy a Simple Contract}
\textbf{Steps:}
\begin{enumerate}
    \item Open Remix IDE: \url{https://remix.ethereum.org}
    \item Create new file: \texttt{SimpleStorage.sol}
    \item Write a storage contract with get/set functions
    \item Compile using Solidity compiler
    \item Deploy to JavaScript VM (local testing)
\end{enumerate}

\vspace{0.3cm}
\textbf{Key Concepts:}
\begin{itemize}
    \item Contract deployment creates new address
    \item Constructor runs once at deployment
    \item State variables persist on blockchain
\end{itemize}
\end{frame}

\begin{frame}{Exercise 2: Read Functions}
\textbf{Characteristics:}
\begin{itemize}
    \item \texttt{view}: Reads state, no modification
    \item \texttt{pure}: No state access, computation only
    \item No gas cost when called externally
    \item Instant response (no transaction needed)
\end{itemize}

\vspace{0.3cm}
\textbf{Tasks:}
\begin{enumerate}
    \item Call \texttt{get()} function to read stored value
    \item Observe: no MetaMask popup, instant result
    \item Note the return value in Remix console
\end{enumerate}
\end{frame}

\begin{frame}{Exercise 3: Write Functions}
\textbf{Characteristics:}
\begin{itemize}
    \item Modify blockchain state
    \item Require gas payment
    \item Create transaction (needs confirmation)
    \item May emit events
\end{itemize}

\vspace{0.3cm}
\textbf{Tasks:}
\begin{enumerate}
    \item Call \texttt{set(42)} to store new value
    \item Observe: transaction created, gas consumed
    \item Verify change by calling \texttt{get()}
    \item Check transaction in Remix terminal
\end{enumerate}
\end{frame}

\begin{frame}{Exercise 4: Events and Logs}
\textbf{Why Events?}
\begin{itemize}
    \item Notify external applications of state changes
    \item Cheaper than storage for historical data
    \item Indexed parameters enable efficient filtering
\end{itemize}

\vspace{0.3cm}
\textbf{Tasks:}
\begin{enumerate}
    \item Add event to contract: \texttt{event ValueChanged(uint newValue)}
    \item Emit event in set function
    \item Redeploy and test
    \item View logs in Remix console
\end{enumerate}
\end{frame}

\begin{frame}{Key Takeaways}
\begin{itemize}
    \item Remix IDE is the standard tool for contract development
    \item View/pure functions are free (no gas)
    \item Write functions cost gas and create transactions
    \item Events provide efficient notification mechanism
    \item Always test on local VM before deploying to testnet
\end{itemize}
