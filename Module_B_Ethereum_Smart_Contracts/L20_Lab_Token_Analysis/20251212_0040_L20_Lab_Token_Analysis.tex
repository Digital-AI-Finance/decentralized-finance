\begin{frame}{Exercise 1: Token Discovery}
\textbf{Tools:}
\begin{itemize}
    \item Etherscan Token Tracker: \url{https://etherscan.io/tokens}
    \item CoinGecko: \url{https://coingecko.com}
    \item DeFiLlama: \url{https://defillama.com}
\end{itemize}

\vspace{0.3cm}
\textbf{Tasks:}
\begin{enumerate}
    \item Browse top ERC-20 tokens by market cap
    \item Select a token for analysis (e.g., LINK, UNI, AAVE)
    \item Note: contract address, total supply, decimals
    \item Check if contract is verified
\end{enumerate}
\end{frame}

\begin{frame}{Exercise 2: Contract Analysis}
\textbf{Key Elements to Check:}
\begin{itemize}
    \item \textbf{Standard Compliance}: Does it implement ERC-20 correctly?
    \item \textbf{Owner Functions}: Can owner mint, pause, blacklist?
    \item \textbf{Transfer Restrictions}: Any hidden fees or taxes?
    \item \textbf{Proxy Pattern}: Is it upgradeable?
\end{itemize}

\vspace{0.3cm}
\textbf{Red Flags:}
\begin{itemize}
    \item Unverified source code
    \item Owner can mint unlimited tokens
    \item Hidden transfer fees
    \item Blacklist functionality without governance
\end{itemize}
\end{frame}

\begin{frame}{Exercise 3: Holder Distribution}
\textbf{Etherscan Holder Analysis:}
\begin{enumerate}
    \item Go to token page -> Holders tab
    \item Note top 10 holders and their percentages
    \item Identify: exchanges, team wallets, contracts
    \item Calculate concentration ratio
\end{enumerate}

\vspace{0.3cm}
\textbf{Healthy Distribution Indicators:}
\begin{itemize}
    \item No single holder > 10\% (except exchanges)
    \item Team tokens locked or vesting
    \item Wide distribution (1000+ unique holders)
    \item Active trading volume
\end{itemize}
\end{frame}

\begin{frame}{Key Takeaways}
\begin{itemize}
    \item Always verify contract source code before interacting
    \item Check holder distribution for concentration risk
    \item Look for red flags: hidden fees, owner privileges
    \item Use multiple tools for comprehensive analysis
    \item High market cap does not guarantee safety
\end{itemize}
