\begin{frame}{Lab Objectives}
\textbf{In this lab, you will:}
\begin{itemize}
\item Select a real cryptocurrency project to analyze

\begin{frame}{Step 4: Calculate Key Metrics}
\textbf{Metric 1: Fully Diluted Valuation (FDV)}
\[
\text{FDV} = \text{Current Price} \times \text{Max Total Supply}
\]

\vspace{0.2cm}
\textbf{Metric 2: FDV/Market Cap Ratio}
\[
\text{Dilution Risk} = \frac{\text{FDV}}{\text{Market Cap}}
\]
\begin{itemize}
\item Ratio > 2: High dilution risk
\item Ratio 1-2: Moderate dilution
\item Ratio close to 1: Low dilution (most supply circulating)
\end{itemize}

\vspace{0.2cm}
\textbf{Metric 3: Network Value to Transactions (NVT)}
\[
\text{NVT} = \frac{\text{Market Cap}}{\text{Daily Transaction Volume (USD)}}
\]

\vspace{0.2cm}
\textbf{Action:} Calculate these metrics for your chosen token.
\end{frame}

\begin{frame}{Step 5: Evaluate Token Classification}
\textbf{Apply the Howey Test:}

\vspace{0.3cm}
\begin{enumerate}
\item \textbf{Investment of Money?}
\begin{itemize}
\item Did investors pay for tokens? (Yes/No)
\end{itemize}

\item \textbf{Common Enterprise?}
\begin{itemize}
\item Are returns pooled/shared? (Yes/No)
\end{itemize}

\item \textbf{Expectation of Profits?}
\begin{itemize}
\item Is it marketed as an investment? (Yes/No)
\end{itemize}

\item \textbf{Efforts of Others?}
\begin{itemize}
\item Do profits depend on team/foundation work? (Yes/No)
\end{itemize}
\end{enumerate}

\vspace{0.3cm}
\textbf{Conclusion:}
\begin{itemize}
\item All four = Security
\item Fails ``efforts of others'' = Likely Utility Token
\end{itemize}

\vspace{0.3cm}
\textbf{Action:} Write 1-2 paragraphs on whether your token is likely a security.
\end{frame}

\begin{frame}{Step 6: Assess Value Accrual Mechanisms}
\textbf{How does the token capture value?}

\vspace{0.3cm}
\textbf{Check for:}
\begin{itemize}
\item \textbf{Fee Sharing:} Do holders receive protocol fees?
\item \textbf{Staking Rewards:} Can tokens be staked for yield?
\item \textbf{Governance:} Voting rights on important decisions?
\item \textbf{Token Burns:} Are tokens permanently removed from supply?
\item \textbf{Utility:} Required to access services?
\item \textbf{Buybacks:} Protocol buying tokens from market?
\end{itemize}

\vspace{0.3cm}
\textbf{Example (Uniswap):}
\begin{itemize}
\item Governance: Yes (vote on protocol changes)
\item Fee sharing: Not yet implemented (controversial)
\item Utility: No fees required to trade
\item Value accrual: Governance power over \$5B+ treasury
\end{itemize}

\vspace{0.3cm}
\textbf{Action:} Identify which mechanisms apply to your token.
\end{frame}

\begin{frame}{Step 7: Analyze Supply Dynamics}
\textbf{Determine supply model:}

\vspace{0.3cm}
\textbf{1. Fixed Supply}
\begin{itemize}
\item Max cap exists
\item Example: Bitcoin (21M), BNB (100M target)
\end{itemize}

\textbf{2. Inflationary}
\begin{itemize}
\item New tokens issued continuously
\item Annual inflation rate: \_\_\_\%
\end{itemize}

\textbf{3. Deflationary}
\begin{itemize}
\item Supply decreases over time
\item Burn mechanism: (describe)
\end{itemize}

\textbf{4. Elastic/Algorithmic}
\begin{itemize}
\item Supply adjusts based on conditions
\item Example: Stablecoins, rebase tokens
\end{itemize}

\vspace{0.3cm}
\textbf{Action:} Classify your token's supply model and justify.
\end{frame}

\begin{frame}{Example Analysis: Uniswap (UNI)}
\textbf{Summary:}
UNI is the governance token of Uniswap, the largest decentralized exchange. Holders vote on protocol upgrades and treasury allocation.

\vspace{0.3cm}
\textbf{Strengths:}
\begin{itemize}
\item Strong product-market fit (billions in daily volume)
\item Decentralized distribution (60\% to community)
\item Governance over massive treasury (\$5B+)
\item Network effects and brand recognition
\end{itemize}

\textbf{Weaknesses:}
\begin{itemize}
\item No direct fee sharing yet (governance only)
\item High FDV/MC ratio (~1.8x dilution)
\item Regulatory uncertainty (SEC scrutiny)
\item Competition from other DEXs
\end{itemize}

\textbf{Verdict:} Hold. Strong fundamentals, but value accrual uncertain until fee switch activated.
\end{frame}

\begin{frame}{Common Pitfalls to Avoid}
\textbf{In your analysis, watch out for:}

\vspace{0.3cm}
\begin{enumerate}
\item \textbf{Relying on old data}
\begin{itemize}
\item Use current market data (not outdated reports)
\end{itemize}

\item \textbf{Ignoring unlock events}
\begin{itemize}
\item Upcoming dilution is critical
\end{itemize}

\item \textbf{Comparing apples to oranges}
\begin{itemize}
\item Only compare similar token types (DeFi to DeFi, L1 to L1)
\end{itemize}

\item \textbf{Focusing only on price}
\begin{itemize}
\item Tokenomics matter more than short-term charts
\end{itemize}

\item \textbf{Ignoring qualitative factors}
\begin{itemize}
\item Team, tech, community are important too
\end{itemize}
\end{enumerate}
\end{frame}
