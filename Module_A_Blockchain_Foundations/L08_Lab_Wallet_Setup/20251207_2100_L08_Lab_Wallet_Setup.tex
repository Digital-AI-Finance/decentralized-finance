\begin{frame}{What is MetaMask?}
\textbf{Overview:}
\begin{itemize}
    \item Browser extension and mobile app cryptocurrency wallet
    \item Supports Ethereum and EVM-compatible blockchains
    \item Non-custodial: you control private keys
    \item Free and open-source
    \item Most popular Ethereum wallet (~30 million users)
\end{itemize}

\vspace{0.3cm}
\textbf{Key Features:}
\begin{itemize}
    \item Send and receive ETH and ERC-20 tokens
    \item Interact with decentralized applications (dApps)
    \item Multiple account management
    \item Network switching (Mainnet, testnets, custom networks)
    \item Hardware wallet integration (Ledger, Trezor)
\end{itemize}

\vspace{0.3cm}
\textbf{Alternatives:}
\begin{itemize}
    \item Browser: Coinbase Wallet, Rabby, Frame
    \item Mobile: Trust Wallet, Rainbow, Argent
    \item Hardware: Ledger, Trezor (highest security)
\end{itemize}
\end{frame}

\begin{frame}{Exercise 1: Installing MetaMask}
\textbf{Installation Steps:}

\vspace{0.3cm}
\begin{enumerate}
    \item Visit official website: \url{https://metamask.io}
    \item Click ``Download'' and select your browser
    \item Install extension from official browser store:
    \begin{itemize}
        \item Chrome: Chrome Web Store
        \item Firefox: Firefox Add-ons
        \item Brave: Chrome Web Store
        \item Edge: Microsoft Edge Add-ons
    \end{itemize}
    \item Pin extension to browser toolbar for easy access
    \item Click MetaMask icon to launch setup wizard
\end{enumerate}

\vspace{0.3cm}
\textbf{Security Warning:}
\begin{itemize}
    \item Only install from official sources
    \item Beware of phishing sites (check URL carefully)
    \item Official website uses HTTPS and \texttt{metamask.io} domain
    \item Fake extensions exist in browser stores (verify developer name)
\end{itemize}
\end{frame}

\begin{frame}{Exercise 2: Creating Your Wallet}
\textbf{Setup Wizard Steps:}

\vspace{0.3cm}
\begin{enumerate}
    \item Click ``Create a new wallet''
    \item Agree to terms of service
    \item Set a strong password (8+ characters, mix of letters/numbers/symbols)
    \item Watch the seed phrase explanation video
    \item Reveal and record your 12-word seed phrase
    \item Confirm seed phrase by selecting words in correct order
    \item Wallet creation complete
\end{enumerate}

\vspace{0.3cm}
\textbf{Password Purpose:}
\begin{itemize}
    \item Encrypts wallet on your device
    \item Required each time you access wallet
    \item Does NOT recover wallet if you forget it
    \item Password protects seed phrase on local device only
\end{itemize}

\vspace{0.3cm}
\textbf{Critical: Seed Phrase vs. Password}
\begin{itemize}
    \item Seed phrase = master backup (recovers wallet on any device)
    \item Password = local device protection (cannot recover wallet)
    \item Lose seed phrase -> lose wallet permanently
\end{itemize}
\end{frame}

\begin{frame}{Seed Phrase Security: Critical Importance}
\textbf{What is a Seed Phrase?}
\begin{itemize}
    \item 12-word mnemonic (BIP-39 standard)
    \item Encodes master private key
    \item Deterministically generates all wallet addresses
    \item Anyone with seed phrase has FULL control of wallet
\end{itemize}

\vspace{0.3cm}
\textbf{Backup Best Practices:}
\begin{enumerate}
    \item Write down on paper (never digital storage)
    \item Verify correct spelling and order
    \item Store in secure physical location (safe, safety deposit box)
    \item Consider multiple redundant copies in separate locations
    \item Never share with anyone (no support staff will ask for it)
    \item Do not store in:
    \begin{itemize}
        \item Email, cloud storage, screenshots, text files
        \item Password managers (controversial, higher risk)
    \end{itemize}
\end{enumerate}

\vspace{0.3cm}
\textbf{Advanced: Steel Backup}
\begin{itemize}
    \item Engrave seed phrase on metal plate (fireproof, waterproof)
    \item Products: Cryptosteel, Billfodl, Blockplate
    \item Protects against physical disasters
\end{itemize}
\end{frame}

\begin{frame}{Exercise 6: Sending Your First Transaction}
\textbf{Transaction Steps:}

\vspace{0.3cm}
\begin{enumerate}
    \item Ensure you have Sepolia ETH (from faucet)
    \item Click ``Send'' button in MetaMask
    \item Enter recipient address:
    \begin{itemize}
        \item Option 1: Create second account in MetaMask (send to yourself)
        \item Option 2: Use classmate's address (coordinate in pairs)
        \item Option 3: Use burn address: \texttt{0x0000...0000} (test only)
    \end{itemize}
    \item Enter amount (e.g., 0.1 Sepolia ETH)
    \item Review transaction details:
    \begin{itemize}
        \item Gas fee estimate (varies by network congestion)
        \item Total amount (amount + gas fee)
    \end{itemize}
    \item Click ``Confirm''
    \item Wait for transaction confirmation (~15 seconds on Sepolia)
    \item View in Activity tab (shows pending -> confirmed)
\end{enumerate}

\vspace{0.3cm}
\textbf{Understanding Gas Fees:}
\begin{itemize}
    \item Gas = computational effort to execute transaction
    \item Fee = gas units $\times$ gas price (in gwei)
    \item Higher gas price = faster confirmation (priority)
    \item Sepolia: fees are negligible (testnet)
    \item Mainnet: fees vary widely (1-50+ USD during congestion)
\end{itemize}
\end{frame}

\begin{frame}{Exercise 8: Creating Additional Accounts}
\textbf{Why Multiple Accounts?}
\begin{itemize}
    \item Organize funds by purpose (personal, business, savings)
    \item Privacy (different addresses not easily linked)
    \item Security (isolate risky activities)
\end{itemize}

\vspace{0.3cm}
\textbf{Creating New Account:}

\vspace{0.3cm}
\begin{enumerate}
    \item Click account menu (circular icon, top-right)
    \item Click ``Add account or hardware wallet''
    \item Select ``Add a new account''
    \item Name account (e.g., ``Account 2'')
    \item New account created with new address
\end{enumerate}

\vspace{0.3cm}
\textbf{Important Notes:}
\begin{itemize}
    \item All accounts derived from same seed phrase
    \item Seed phrase recovers ALL accounts
    \item Accounts use BIP-44 derivation path: \texttt{m/44'/60'/0'/0/index}
    \item Switching accounts: click account menu, select account
\end{itemize}

\vspace{0.3cm}
\textbf{Task:}
\begin{itemize}
    \item Create second account
    \item Send Sepolia ETH from Account 1 to Account 2
    \item Verify transaction in block explorer
\end{itemize}
\end{frame}

\begin{frame}{Security Best Practices}
\textbf{Critical Security Rules:}

\vspace{0.3cm}
\begin{enumerate}
    \item \textbf{Seed Phrase:}
    \begin{itemize}
        \item Never share with anyone (no exceptions)
        \item Never enter on websites (phishing)
        \item Never store digitally (paper only)
        \item Never photograph or screenshot
    \end{itemize}

    \item \textbf{Phishing Protection:}
    \begin{itemize}
        \item Bookmark MetaMask extension (avoid typing URL)
        \item Verify website URLs before connecting wallet
        \item Beware of fake support contacts (MetaMask never DMs first)
        \item Check transaction details before confirming
    \end{itemize}

    \item \textbf{Network Verification:}
    \begin{itemize}
        \item Always confirm correct network before transacting
        \item Mainnet transactions use real money (double-check)
        \item Testnet assets have no value (safe to experiment)
    \end{itemize}

    \item \textbf{Transaction Approval:}
    \begin{itemize}
        \item Review recipient address carefully (irreversible)
        \item Check amount and gas fees
        \item Start with small test transactions
        \item Revoke unnecessary token approvals (debank.com, revoke.cash)
    \end{itemize}
\end{enumerate}
\end{frame}

\begin{frame}{Common Security Threats}
\textbf{Attack Vectors:}

\vspace{0.3cm}
\begin{enumerate}
    \item \textbf{Phishing Websites:}
    \begin{itemize}
        \item Fake dApp sites request seed phrase
        \item Look-alike domains (metamask.com vs metarnask.com)
        \item Mitigation: bookmark trusted sites, verify URLs
    \end{itemize}

    \item \textbf{Malicious dApps:}
    \begin{itemize}
        \item Request unlimited token spending approval
        \item Drain wallet after approval
        \item Mitigation: review permissions, limit approvals
    \end{itemize}

    \item \textbf{Clipboard Malware:}
    \begin{itemize}
        \item Replaces copied address with attacker's address
        \item Victim sends funds to wrong address
        \item Mitigation: verify first/last characters after pasting
    \end{itemize}

    \item \textbf{Fake Support Scams:}
    \begin{itemize}
        \item Impersonators offer help via DM
        \item Ask for seed phrase or remote access
        \item Mitigation: never respond, MetaMask has no live support DMs
    \end{itemize}
\end{enumerate}
\end{frame}
