\begin{frame}{What is MetaMask?}
\textbf{Overview:}
\begin{itemize}
    \item Browser extension and mobile app cryptocurrency wallet
    \item Supports Ethereum and EVM-compatible blockchains
    \item Non-custodial: you control private keys
    \item Most popular Ethereum wallet (~30 million users)
\end{itemize}

\vspace{0.3cm}
\textbf{Key Features:}
\begin{itemize}
    \item Send and receive ETH and ERC-20 tokens
    \item Interact with decentralized applications (dApps)
    \item Multiple account management
    \item Network switching (Mainnet, testnets, custom networks)
\end{itemize}
\end{frame}

\begin{frame}{Exercise 1: Installing MetaMask}
\textbf{Installation Steps:}
\begin{enumerate}
    \item Visit official website: \url{https://metamask.io}
    \item Click ``Download'' and select your browser
    \item Install extension from official browser store
    \item Pin extension to browser toolbar
    \item Click MetaMask icon to launch setup wizard
\end{enumerate}

\vspace{0.3cm}
\textbf{Security Warning:}
\begin{itemize}
    \item Only install from official sources
    \item Beware of phishing sites (check URL carefully)
    \item Fake extensions exist in browser stores
\end{itemize}
\end{frame}

\begin{frame}{Exercise 2: Creating Your Wallet}
\textbf{Setup Wizard Steps:}
\begin{enumerate}
    \item Click ``Create a new wallet''
    \item Set a strong password (8+ characters)
    \item Reveal and record your 12-word seed phrase
    \item Confirm seed phrase by selecting words in correct order
\end{enumerate}

\vspace{0.3cm}
\textbf{Critical: Seed Phrase vs. Password}
\begin{itemize}
    \item Seed phrase = master backup (recovers wallet on any device)
    \item Password = local device protection (cannot recover wallet)
    \item \textbf{Lose seed phrase -> lose wallet permanently}
\end{itemize}
\end{frame}

\begin{frame}{Exercise 3: Adding Sepolia Testnet}
\textbf{Why Use Testnets?}
\begin{itemize}
    \item Practice without risking real money
    \item Free testnet ETH available from faucets
    \item Identical functionality to mainnet
\end{itemize}

\vspace{0.3cm}
\textbf{Adding Sepolia Network:}
\begin{enumerate}
    \item Click network selector (top-right dropdown)
    \item Settings -> Advanced -> Show test networks (enable)
    \item Select ``Sepolia test network''
\end{enumerate}

\vspace{0.3cm}
\textbf{Sepolia Faucets:}
\begin{itemize}
    \item Alchemy: \url{https://sepoliafaucet.com}
    \item Infura: \url{https://www.infura.io/faucet/sepolia}
\end{itemize}
\end{frame}

\begin{frame}{Exercise 4: Sending Your First Transaction}
\textbf{Transaction Steps:}
\begin{enumerate}
    \item Ensure you have Sepolia ETH (from faucet)
    \item Click ``Send'' button in MetaMask
    \item Enter recipient address (create second account or use classmate's)
    \item Enter amount (e.g., 0.1 Sepolia ETH)
    \item Review transaction details (gas fee, total)
    \item Click ``Confirm''
    \item View in Activity tab
\end{enumerate}

\vspace{0.3cm}
\textbf{Understanding Gas Fees:}
\begin{itemize}
    \item Fee = gas units $\times$ gas price (in gwei)
    \item Sepolia: fees are negligible (testnet)
\end{itemize}
\end{frame}

\begin{frame}{Security Best Practices}
\textbf{Critical Security Rules:}
\begin{enumerate}
    \item \textbf{Seed Phrase:} Never share, never store digitally
    \item \textbf{Phishing Protection:} Verify URLs, bookmark trusted sites
    \item \textbf{Network Verification:} Confirm correct network before transacting
    \item \textbf{Transaction Approval:} Review recipient and amounts carefully
\end{enumerate}

\vspace{0.3cm}
\textbf{Common Threats:}
\begin{itemize}
    \item Phishing websites requesting seed phrase
    \item Malicious dApps with unlimited token approvals
    \item Clipboard malware replacing copied addresses
    \item Fake support scams via DM
\end{itemize}
\end{frame}
