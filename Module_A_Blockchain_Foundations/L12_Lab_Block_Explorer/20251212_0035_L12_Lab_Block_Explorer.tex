\begin{frame}{Exercise 1: Analyzing Bitcoin Transactions}
\textbf{Instructions:}
\begin{enumerate}
    \item Visit: \url{https://blockstream.info}
    \item Search for the first-ever Bitcoin transaction:\\
    \texttt{f4184fc596403b9d638783cf57adfe4c75c605f6356fbc91338530e9831e9e16}
    \item Examine: inputs, outputs, fee, block number, confirmations
\end{enumerate}

\vspace{0.3cm}
\textbf{Key Observations:}
\begin{itemize}
    \item \textbf{Inputs:} Reference to previous transaction output
    \item \textbf{Outputs:} New UTXOs with amounts and addresses
    \item \textbf{Fee:} Difference between input sum and output sum
    \item \textbf{Confirmations:} Blocks built on top of this transaction
\end{itemize}
\end{frame}

\begin{frame}{Exercise 2: Analyzing Ethereum Transactions}
\textbf{Instructions:}
\begin{enumerate}
    \item Visit: \url{https://etherscan.io}
    \item Select a recent transaction from homepage
    \item Examine: hash, status, block, from, to, value, gas
\end{enumerate}

\vspace{0.3cm}
\textbf{Gas Mechanics:}
\begin{itemize}
    \item Gas Limit: Maximum gas user willing to pay
    \item Gas Used: Actual gas consumed
    \item Gas Price: Price per unit (in gwei)
    \item Transaction Fee: Gas Used $\times$ Gas Price
\end{itemize}

\vspace{0.3cm}
\textbf{Example:} 21,000 gas $\times$ 32 gwei = 0.000672 ETH
\end{frame}

\vspace{0.3cm}
\begin{enumerate}
    \item \textbf{Exchange Address:} High volume, large balances, labeled

    \item \textbf{Individual User:} Low volume, smaller balances

\begin{frame}{Exercise 4: Forensic Case Study}
\textbf{Scenario:} Tracking fund flows

\vspace{0.3cm}
\textbf{Investigation Steps:}
\begin{enumerate}
    \item Confirm source transaction
    \item Click on destination address
    \item View subsequent transactions
    \item Follow the chain of transfers
    \item Document path with transaction hashes
\end{enumerate}

\vspace{0.3cm}
\textbf{Privacy Techniques to Recognize:}
\begin{itemize}
    \item Mixers/Tumblers (e.g., Tornado Cash)
    \item CoinJoin transactions
    \item Exchange deposits (off-chain after deposit)
\end{itemize}
\end{frame}

\begin{frame}{Key Takeaways}
\begin{itemize}
    \item Block explorers provide transparency into all blockchain activity
    \item Bitcoin uses UTXO model; Ethereum uses account model
    \item Gas fees vary based on network congestion
    \item Addresses can be identified by transaction patterns
    \item Blockchain forensics is powerful but faces privacy challenges
    \item Verified smart contracts allow code auditing
\end{itemize}

\vspace{0.5cm}
\textbf{Real-World Applications:}
\begin{itemize}

    \item Fraud investigation
