\begin{frame}{Exercise 1: Project Research}
\textbf{Research Checklist:}
\begin{itemize}
    \item Team: Doxxed? Track record? LinkedIn profiles?
    \item Roadmap: Realistic? Delivered milestones?
    \item Community: Discord activity? Twitter engagement?
    \item Partnerships: Verified? Real companies?
\end{itemize}

\vspace{0.3cm}
\textbf{Red Flags:}
\begin{itemize}
    \item Anonymous team with no track record
    \item Unrealistic promises (guaranteed profits)
    \item Fake partnerships or celebrity endorsements
    \item Copied art or unoriginal concept
\end{itemize}
\end{frame}

\begin{frame}{Exercise 2: On-Chain Analysis}
\textbf{Etherscan Analysis:}
\begin{enumerate}
    \item Contract address verification
    \item Owner/admin functions review
    \item Mint function (max supply enforced?)
    \item Transfer restrictions
\end{enumerate}

\vspace{0.3cm}
\textbf{Holder Analysis:}
\begin{itemize}
    \item Distribution across wallets
    \item Team wallet identification
    \item Whale concentration
    \item Trading patterns (organic vs wash)
\end{itemize}
\end{frame}

\begin{frame}{Exercise 3: Risk Assessment}
\textbf{Risk Categories:}
\begin{enumerate}
    \item \textbf{Liquidity Risk}: Can you sell when needed?
    \item \textbf{Smart Contract Risk}: Bugs, exploits, centralization
    \item \textbf{Market Risk}: Volatility, bear markets
    \item \textbf{Rug Pull Risk}: Team abandonment, exit scam
    \item \textbf{Platform Risk}: Marketplace dependency
\end{enumerate}

\vspace{0.3cm}
\textbf{Create Risk Score:}
\begin{itemize}
    \item Rate each category 1-10
    \item Calculate weighted average
    \item Determine: Low (<4), Medium (4-7), High (>7)
\end{itemize}
\end{frame}

\begin{frame}{Key Takeaways}
\begin{itemize}
    \item Always DYOR (Do Your Own Research)
    \item Team reputation is critical - anonymous = higher risk
    \item On-chain analysis reveals true holder behavior
    \item Multiple risk factors compound overall risk
    \item Most NFT projects fail - invest accordingly
\end{itemize}
