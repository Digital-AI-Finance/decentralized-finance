\begin{frame}{Comprehensive Due Diligence Framework}
\textbf{8-Pillar Evaluation Model:}

\vspace{0.3cm}
\begin{enumerate}
\item \textbf{Technology \& Infrastructure} (L21-22)
\item \textbf{Marketplace Presence \& Liquidity} (L23-24)
\item \textbf{Community \& Social Signals} (L25)
\item \textbf{Team \& Governance}
\item \textbf{Utility \& Roadmap}
\item \textbf{Rarity \& Distribution}
\item \textbf{Financial Metrics}
\item \textbf{Legal \& Regulatory Compliance} (L27)
\end{enumerate}

\vspace{0.3cm}
\textbf{Scoring:} Each pillar scored 0-10, total score out of 80
\end{frame}

\begin{frame}{Pillar 1: Technology \& Infrastructure}
\textbf{Key Questions:}

\vspace{0.3cm}
\begin{itemize}
\item Is the smart contract verified on Etherscan?
\item Does it use standard ERC-721 or ERC-1155 (or custom)?
\item Is metadata stored on IPFS, Arweave, or centralized server?
\item Is the \texttt{tokenURI} immutable or mutable?
\item Are images pinned with reputable service?
\item Are there any smart contract security audits?
\item Is the contract ownership renounced?
\end{itemize}

\vspace{0.3cm}
\textbf{Scoring (0-10):}
\begin{itemize}
\item 10: Verified, standard, IPFS/Arweave, immutable, audited, renounced
\item 5: Verified, custom contract, IPFS, some mutability
\item 0: Unverified or HTTP metadata or active admin control
\end{itemize}
\end{frame}

\begin{frame}{Pillar 2: Marketplace Presence \& Liquidity}
\textbf{Key Metrics:}

\vspace{0.3cm}
\begin{itemize}
\item Floor price (ETH) and stability over 30 days
\item 7-day trading volume (ETH)
\item Number of unique buyers in past 30 days
\item Percentage of collection listed for sale
\item Average sale price vs. floor price (premium analysis)
\item Presence on multiple marketplaces (OpenSea, Blur, etc.)
\item Wash trading indicators (volume vs. unique traders)
\end{itemize}

\vspace{0.3cm}
\textbf{Scoring (0-10):}
\begin{itemize}
\item 10: >10\% listed, daily sales, >100 unique buyers/month, multi-platform
\item 5: 5-10\% listed, weekly sales, 20-100 buyers/month
\item 0: <2\% listed, <5 sales/month, <10 buyers (illiquid)
\end{itemize}
\end{frame}

\begin{frame}{Pillar 3: Community \& Social Signals}
\textbf{Evaluation Criteria:}

\vspace{0.3cm}
\begin{itemize}
\item Twitter followers and engagement rate
\item Discord members and daily active users
\item Reddit/Telegram community presence
\item Organic vs. bot followers (check engagement ratio)
\item Influencer and celebrity holders
\item Community sentiment (bullish, neutral, bearish)
\item Recent communication frequency (active vs. silent)
\end{itemize}

\vspace{0.3cm}
\textbf{Scoring (0-10):}
\begin{itemize}
\item 10: >50k Twitter, >10k Discord, daily updates, high engagement
\item 5: 10-50k Twitter, 2-10k Discord, weekly updates
\item 0: <5k Twitter, inactive Discord, no recent updates (dead community)
\end{itemize}
\end{frame}

\begin{frame}{Pillar 4: Team \& Governance}
\textbf{Key Questions:}

\vspace{0.3cm}
\begin{itemize}
\item Is the team doxxed (real identities public)?
\item What is the team's track record (prior projects)?
\item Is there a DAO or governance structure?
\item What percentage of supply does the team hold?
\item Have there been any controversies or rug pulls?
\item Are advisors or investors credible (VCs, known figures)?
\end{itemize}

\vspace{0.3cm}
\textbf{Scoring (0-10):}
\begin{itemize}
\item 10: Fully doxxed, proven track record, <10\% team holdings, DAO
\item 5: Pseudonymous but credible, 10-20\% team holdings
\item 0: Anonymous, no track record, >30\% team holdings (rug risk)
\end{itemize}
\end{frame}

\begin{frame}{Pillar 5: Utility \& Roadmap}
\textbf{Utility Categories:}

\vspace{0.3cm}
\begin{itemize}
\item \textbf{IP Rights:} Commercial usage rights (BAYC model)
\item \textbf{Ecosystem Access:} Token-gated content, events, products
\item \textbf{Governance:} Voting on project decisions
\item \textbf{Yield:} Staking rewards or passive income
\item \textbf{Metaverse:} Integration with virtual worlds
\item \textbf{Gaming:} Playable characters or in-game items
\end{itemize}

\vspace{0.3cm}
\textbf{Roadmap Execution:}
\begin{itemize}
\item Have past milestones been delivered on time?
\item Are future goals realistic and specific?
\item Is the roadmap vague or overpromising?
\end{itemize}

\vspace{0.3cm}
\textbf{Scoring (0-10):}
\begin{itemize}
\item 10: Multiple utility streams, strong roadmap execution
\item 5: Some utility, mixed execution
\item 0: No utility beyond speculation, failed roadmap
\end{itemize}
\end{frame}

\begin{frame}{Pillar 6: Rarity \& Distribution}
\textbf{Rarity Analysis:}

\vspace{0.3cm}
\begin{itemize}
\item Use rarity.tools or similar to rank NFTs
\item Identify rarest traits (top 5\% frequency)
\item Check if rarity premium exists (rare items sell for higher multiples)
\item Analyze trait distribution (balanced vs. skewed)
\end{itemize}

\vspace{0.3cm}
\textbf{Holder Distribution:}
\begin{itemize}
\item Unique holders / Total supply ratio (target >60\%)
\item Top 10 holders ownership percentage (target <20\%)
\item Whale concentration (any holder with >10\%?)
\item Team and founder holdings
\end{itemize}

\vspace{0.3cm}
\textbf{Scoring (0-10):}
\begin{itemize}
\item 10: Clear rarity structure, >70\% holder ratio, no whales
\item 5: Moderate rarity, 50-70\% holder ratio
\item 0: No rarity premium, <40\% holder ratio (concentrated)
\end{itemize}
\end{frame}

\begin{frame}{Pillar 7: Financial Metrics}
\textbf{Key Calculations:}

\vspace{0.3cm}
\begin{enumerate}
\item \textbf{Market Cap:} Floor Price * Total Supply
\item \textbf{Volume/MCap Ratio:} 30d Volume / Market Cap (liquidity)
\item \textbf{Mint vs. Floor:} Floor price relative to original mint
\item \textbf{ATH Distance:} Current floor vs. all-time high
\item \textbf{Holder ROI:} Average holder return (current floor vs. avg buy)
\end{enumerate}

\vspace{0.3cm}
\textbf{Example:}
\begin{itemize}
\item Floor: 2 ETH, Supply: 10,000 = Market Cap 20,000 ETH
\item 30d Volume: 1,000 ETH / 20,000 ETH = 5\% (moderate liquidity)
\item Mint: 0.08 ETH, Floor: 2 ETH = 25x return (strong)
\end{itemize}

\vspace{0.3cm}
\textbf{Scoring (0-10):}
\begin{itemize}
\item 10: High liquidity, strong mint ROI, near ATH
\item 5: Moderate liquidity, breakeven mint ROI
\item 0: Illiquid, below mint price (underwater holders)
\end{itemize}
\end{frame}

\begin{frame}{Pillar 8: Legal \& Regulatory Compliance}
\textbf{Key Questions:}

\vspace{0.3cm}
\begin{itemize}
\item Are IP rights clearly defined in legal terms?
\item Does the project have Terms of Service?
\item Is there a legal entity (DAO LLC, foundation)?
\item Are there potential securities law issues (Howey Test)?
\item Does the project face regulatory scrutiny?
\item Are there trademark or copyright disputes?
\end{itemize}

\vspace{0.3cm}
\textbf{Scoring (0-10):}
\begin{itemize}
\item 10: Clear IP rights, legal entity, compliant, no disputes
\item 5: Some legal clarity, no major issues
\item 0: Vague IP, no legal structure, securities risk, ongoing lawsuits
\end{itemize}
\end{frame}

\begin{frame}{Red Flags Identification}
\textbf{Critical Red Flags (Automatic Disqualification):}

\vspace{0.3cm}
\begin{enumerate}
\item \textbf{Unverified contract:} Cannot audit code
\item \textbf{HTTP metadata:} Centralized, loss risk
\item \textbf{Team holds >50\%:} Dump risk
\item \textbf{No community activity:} Dead project
\item \textbf{Zero sales in 30 days:} Illiquid
\item \textbf{Suspected rug pull history:} Team scammed before
\end{enumerate}

\vspace{0.3cm}
\textbf{Warning Flags (Reduce Score):}
\begin{itemize}
\item Mutable \texttt{tokenURI} (metadata can change)
\item High team holdings (20-50\%)
\item Failed roadmap milestones
\item Wash trading indicators
\item Celebrity pump without substance
\end{itemize}

\vspace{0.3cm}
\textbf{Rule:} 2+ critical red flags = DO NOT INVEST
\end{frame}

\begin{frame}{Activity 3: Pillar-by-Pillar Scoring}
\textbf{Task:} Score each of the 8 pillars (0-10)

\vspace{0.3cm}
\textbf{Scoring Process:}
\begin{enumerate}
\item Review collected data for Pillar 1 (Technology)
\item Compare against scoring criteria (slide 4)
\item Assign score 0-10 with justification
\item Repeat for all 8 pillars
\item Calculate total score (out of 80)
\end{enumerate}

\vspace{0.3cm}
\textbf{Interpretation:}
\begin{itemize}
\item 70-80: Excellent project, strong buy candidate
\item 55-69: Solid project, acceptable risk
\item 40-54: High risk, speculative
\item $<$40: Avoid, too many weaknesses
\end{itemize}
\end{frame}

\begin{frame}{Activity 4: Comparative Analysis}
\textbf{Task:} Compare your selected collection to 2 competitors

\vspace{0.3cm}
\textbf{Steps:}
\begin{enumerate}
\item Identify 2 similar collections (same category, similar floor)
\item Score both competitors using same 8-pillar framework
\item Create comparison table with all scores
\item Analyze relative strengths and weaknesses
\item Determine which offers best value proposition
\end{enumerate}

\vspace{0.3cm}
\textbf{Example Comparison:}
\begin{itemize}
\item Your collection: 62/80 (floor 2 ETH)
\item Competitor A: 58/80 (floor 2.5 ETH) -- Overvalued
\item Competitor B: 67/80 (floor 1.8 ETH) -- Undervalued
\end{itemize}

\vspace{0.3cm}
\textbf{Conclusion:} Competitor B may be better value
\end{frame}

\begin{frame}{Activity 5: Risk Assessment}
\textbf{Task:} Identify and quantify all risks

\vspace{0.3cm}
\textbf{Risk Categories:}
\begin{enumerate}
\item \textbf{Technology Risk:} Smart contract bugs, metadata loss
\item \textbf{Market Risk:} Liquidity dries up, floor price crashes
\item \textbf{Team Risk:} Abandonment, rug pull, mismanagement
\item \textbf{Regulatory Risk:} Securities classification, IP disputes
\item \textbf{Competition Risk:} Better projects emerge, market share loss
\end{enumerate}

\vspace{0.3cm}
\textbf{Risk Matrix:}
\begin{itemize}
\item High Probability + High Impact = Critical Risk
\item High Probability + Low Impact = Monitor
\item Low Probability + High Impact = Contingency Plan
\item Low Probability + Low Impact = Accept
\end{itemize}
\end{frame}

\begin{frame}{Activity 6: Investment Thesis Development}
\textbf{Task:} Articulate clear bull and bear cases

\vspace{0.3cm}
\textbf{Bull Case (Why Buy):}
\begin{itemize}
\item Strengths from 8-pillar analysis
\item Undervalued relative to competitors
\item Strong roadmap execution potential
\item Growing community and ecosystem
\item Catalysts (upcoming launches, partnerships)
\end{itemize}

\vspace{0.3cm}
\textbf{Bear Case (Why Avoid):}
\begin{itemize}
\item Weaknesses from 8-pillar analysis
\item Overvalued relative to fundamentals
\item Market saturation in category
\item Regulatory uncertainty
\item Critical dependencies (team, platform)
\end{itemize}

\vspace{0.3cm}
\textbf{Balanced Thesis:} Weight both cases, determine net conviction
\end{frame}

\begin{frame}{Activity 7: Investment Recommendation}
\textbf{Task:} Provide final recommendation with rationale

\vspace{0.3cm}
\textbf{Recommendation Types:}
\begin{enumerate}
\item \textbf{Strong Buy:} Score 70+, undervalued, low risk
\item \textbf{Buy:} Score 55-69, fair value, acceptable risk
\item \textbf{Hold:} Score 40-54, fair value, monitoring needed
\item \textbf{Sell/Avoid:} Score <40 or critical red flags
\end{enumerate}

\vspace{0.3cm}
\textbf{Investment Parameters:}
\begin{itemize}
\item Recommended position size (e.g., 1-5\% of portfolio)
\item Target entry price (floor or below)
\item Stop-loss (e.g., 30\% below entry)
\item Time horizon (short-term flip vs. long-term hold)
\item Exit strategy (take profit at X\% gain or milestones)
\end{itemize}
\end{frame}

\begin{frame}{Common Pitfalls to Avoid}
\textbf{Analytical Errors:}

\vspace{0.3cm}
\begin{enumerate}
\item \textbf{Overweighting hype:} Celebrity endorsements $\neq$ fundamentals
\item \textbf{Ignoring liquidity:} High floor but no buyers = trap
\item \textbf{Recency bias:} Recent pump does not mean sustainable growth
\item \textbf{Confirmation bias:} Seeking data that supports pre-existing opinion
\item \textbf{Anchoring:} Fixating on ATH or mint price (irrelevant)
\item \textbf{FOMO:} Buying at peak hype without analysis
\end{enumerate}

\vspace{0.3cm}
\textbf{Best Practices:}
\begin{itemize}
\item Base analysis on objective data, not emotions
\item Validate social metrics (bots vs. real engagement)
\item Always check smart contract and metadata
\item Compare to fundamentals, not just price action
\end{itemize}
\end{frame}

\begin{itemize}

\end{itemize}

\vspace{0.3cm}
\textbf{Key Insight:} On-chain data reveals true investor behavior (not just social hype)
\end{frame}
