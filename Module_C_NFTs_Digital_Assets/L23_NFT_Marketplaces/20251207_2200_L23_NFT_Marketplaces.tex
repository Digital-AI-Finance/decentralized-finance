\begin{frame}{NFT Marketplace Landscape}
\textbf{Major Platforms (by Volume, 2024):}

\vspace{0.3cm}
\begin{enumerate}
\item \textbf{OpenSea:} Largest NFT marketplace, multi-chain support
\item \textbf{Blur:} Pro trader-focused, zero marketplace fees
\item \textbf{Magic Eden:} Solana-native, expanding to Ethereum
\item \textbf{Rarible:} Community-governed, aggregator model
\item \textbf{LooksRare:} Rewards-based model, vampire attack on OpenSea
\item \textbf{X2Y2:} Low fees, trader incentives

\vspace{0.3cm}
\textbf{Market Consolidation:}
\begin{itemize}
\item OpenSea and Blur dominate Ethereum NFT trading (>80\% volume)
\item Smaller marketplaces compete on fees, rewards, and niche communities
\end{itemize}

\begin{frame}{OpenSea: The Market Leader}
\textbf{Founded:} 2017, valued at \$13.3B (2022 peak)

\vspace{0.3cm}
\textbf{Key Features:}
\begin{itemize}
\item Multi-chain support (Ethereum, Polygon, Arbitrum, Optimism, etc.)
\item User-friendly interface for creators and collectors
\item Collection verification and curation
\item Lazy minting (create NFTs without upfront gas costs)
\end{itemize}

\vspace{0.3cm}
\textbf{Business Model:}
\begin{itemize}
\item \textbf{Marketplace fee:} 2.5\% on each sale (historically, now optional 0.5\%)
\item \textbf{Creator royalties:} Optional enforcement (formerly mandatory)
\item \textbf{Revenue:} \$2.7B cumulative fees (as of 2024)
\end{itemize}

\vspace{0.3cm}
\textbf{Challenges:} Competition from zero-fee marketplaces, royalty enforcement debate
\end{frame}

\begin{frame}{Blur: The Pro Trader Platform}
\textbf{Launched:} October 2022, rapid growth via airdrop incentives

\vspace{0.3cm}
\textbf{Key Features:}
\begin{itemize}
\item \textbf{Zero marketplace fees:} No commission on trades
\item \textbf{Advanced trading tools:} Portfolio management, sweeping, sniping
\item \textbf{Bid pools:} Aggregated liquidity for collection-wide offers
\item \textbf{Airdrop rewards:} BLUR token distributed to active traders
\end{itemize}

\vspace{0.3cm}
\textbf{Business Model:}
\begin{itemize}
\item Revenue from optional creator royalties (0.5\% default)
\item Platform sustainability via BLUR token value
\end{itemize}

\vspace{0.3cm}
\textbf{Impact:} Surpassed OpenSea in volume during 2023, forced industry fee reductions
\end{frame}

\item \textbf{Aggregator model:} Sources listings from multiple marketplaces
\item \textbf{Creator tools:} Easy NFT creation and minting
\item \textbf{Community curation:} User-driven collection discovery
\end{itemize}

\vspace{0.3cm}
\textbf{Business Model:}
\begin{itemize}
\item 1-2.5\% marketplace fee (varies by chain)
\item Revenue sharing with RARI holders
\end{itemize}

\vspace{0.3cm}
\textbf{Differentiation:} Focus on creator empowerment and decentralization
\end{frame}

\begin{frame}{Listing Mechanics: Fixed Price vs. Auction}
\textbf{Fixed Price Listings:}
\begin{itemize}
\item Seller sets a ``Buy Now'' price
\item Buyer pays the listed price to complete purchase
\item Instant settlement (no waiting period)
\item Most common listing type (>90\% of listings)
\end{itemize}

\vspace{0.3cm}
\textbf{Auction Listings:}
\begin{itemize}
\item \textbf{English Auction:} Ascending bids, highest bidder wins
\item \textbf{Dutch Auction:} Descending price until buyer accepts
\item \textbf{Reserve price:} Minimum acceptable bid
\item Used for high-value or uncertain pricing
\end{itemize}

\vspace{0.3cm}
\textbf{Offer System:}
\begin{itemize}
\item Buyers submit offers below listing price
\item Seller can accept, counter, or ignore
\item Common for negotiation and floor sniping
\end{itemize}
\end{frame}

\begin{frame}{Order Book and Off-Chain Matching}
\textbf{Traditional Order Book:}
\begin{itemize}
\item Centralized database of buy/sell orders
\item Matching engine pairs buyers and sellers
\item Settlement on-chain when trade executes
\end{itemize}

\vspace{0.3cm}
\textbf{OpenSea Seaport Protocol:}
\begin{itemize}
\item \textbf{Off-chain orders:} Listings signed off-chain (no gas cost)
\item \textbf{On-chain settlement:} Trade executes when matched
\item \textbf{Advanced features:} Partial fills, bundle trades, criteria-based orders
\item \textbf{Gas savings:} Sellers pay no gas until sale completes
\end{itemize}

\vspace{0.3cm}
\textbf{Blur's Bid Pools:}
\begin{itemize}
\item Aggregated collection-wide offers
\item Buyers bid on any NFT in a collection (not specific token ID)
\item Sellers can accept best available bid instantly
\end{itemize}
\end{frame}

\begin{frame}{The Royalty Enforcement Debate}
\textbf{Creator Royalties:} Percentage paid to original creator on secondary sales

\vspace{0.3cm}
\textbf{Historical Model (2017-2022):}
\begin{itemize}
\item Marketplaces enforced creator-set royalties (2-10\% typical)
\item Revenue stream for artists and projects
\item Industry standard across all platforms
\end{itemize}

\vspace{0.3cm}
\textbf{Shift to Optional Royalties (2022+):}
\begin{itemize}
\item Blur launched with optional royalties (0.5\% default)
\item X2Y2 and Magic Eden followed suit
\item OpenSea capitulated in 2023 (optional royalties)
\end{itemize}

\vspace{0.3cm}
\textbf{Arguments For Optional:}
\begin{itemize}
\item Buyer freedom, lower transaction costs
\item Market competition drives efficiency
\end{itemize}

\vspace{0.3cm}
\textbf{Arguments Against:}
\begin{itemize}
\item Undermines creator compensation model
\item Reduces incentive for quality projects
\end{itemize}
\end{frame}

\begin{frame}{On-Chain Royalty Enforcement}
\textbf{Technical Challenge:} Smart contracts cannot force royalty payments

\vspace{0.3cm}
\textbf{Why Royalties Are Not Enforceable:}
\begin{itemize}
\item ERC-721 standard has no royalty mechanism
\item Buyers can transfer NFTs directly (peer-to-peer) without marketplace
\item Marketplaces voluntarily honor royalties (not protocol-enforced)
\end{itemize}

\vspace{0.3cm}
\textbf{ERC-2981 Royalty Standard:}
\begin{itemize}
\item On-chain royalty information (creator address, percentage)
\item Marketplaces can query but not forced to enforce
\item Adoption: Widespread but ineffective without marketplace cooperation
\end{itemize}

\vspace{0.3cm}
\textbf{Experimental Solutions:}
\begin{itemize}
\item \textbf{Operator Filter Registry:} Blocks transfers via non-compliant marketplaces
\item \textbf{ERC-721C:} Programmable transfer restrictions in contract
\item Trade-off: Reduces NFT liquidity and composability
\end{itemize}
\end{frame}

\begin{frame}{Wash Trading: Artificial Volume}
\textbf{Wash Trading:} Buyer and seller are the same entity (self-trading)

\vspace{0.3cm}
\textbf{Motivation:}
\begin{itemize}
\item Inflate collection volume and visibility
\item Manipulate floor price upward
\item Farm marketplace reward tokens (e.g., LooksRare, X2Y2)
\item Create false demand signals
\end{itemize}

\vspace{0.3cm}
\textbf{Mechanics:}
\begin{enumerate}
\item Trader owns NFT in Wallet A
\item Trader lists NFT for sale on marketplace
\item Trader buys own NFT using Wallet B
\item Net cost: Marketplace fees + gas (often subsidized by rewards)
\item Repeat to accumulate volume and rewards
\end{enumerate}

\vspace{0.3cm}
\textbf{Detection:}
\begin{itemize}
\item Blockchain analysis reveals wallet connections
\item High transfer frequency between same addresses
\item Volume spikes without genuine collector interest
\end{itemize}
\end{frame}

\begin{frame}{Wash Trading Case Study: LooksRare (2022)}
\textbf{LooksRare Launch Strategy:}
\begin{itemize}
\item Reward traders with LOOKS tokens based on volume
\item Incentivize switching from OpenSea (vampire attack)
\end{itemize}

\vspace{0.3cm}
\textbf{Unintended Consequence:}
\begin{itemize}
\item Traders wash traded to farm LOOKS rewards
\item Daily volume surged to \$400M+ (mostly wash trading)
\item Organic volume remained low (<10\% of reported volume)
\end{itemize}

\vspace{0.3cm}
\textbf{Analysis (Chainalysis, 2022):}
\begin{itemize}
\item Estimated 95\% of LooksRare volume was wash trading
\item Single wallet traded \$23M in one day (same NFT repeatedly)
\end{itemize}

\vspace{0.3cm}
\textbf{Lesson:} Volume-based rewards create perverse incentives for manipulation
\end{frame}

\begin{frame}{Detecting Wash Trading}
\textbf{Red Flags:}

\vspace{0.3cm}
\begin{enumerate}
\item \textbf{High volume, low unique buyers:} Few wallets trading repeatedly
\item \textbf{Same NFTs trading back and forth:} Token ID sold multiple times/day
\item \textbf{Wallet clustering:} Funded by same source address
\item \textbf{Unprofitable trades:} Selling below purchase price + fees
\item \textbf{Reward farming patterns:} Activity spikes during reward periods
\end{enumerate}

\vspace{0.3cm}
\textbf{Analysis Tools:}
\begin{itemize}
\item Nansen: Wallet labeling and flow analysis
\item Dune Analytics: Custom SQL queries on marketplace data
\item Chainalysis: Professional-grade blockchain forensics
\end{itemize}

\vspace{0.3cm}
\textbf{Mitigation:}
\begin{itemize}
\item Focus on unique buyers and holder distribution
\item Verify organic community engagement (social media, Discord)
\end{itemize}
\end{frame}

\begin{frame}{NFT Aggregators}
\textbf{Aggregator:} Platform that sources listings from multiple marketplaces

\vspace{0.3cm}
\textbf{Major Aggregators:}
\begin{itemize}
\item \textbf{Gem (acquired by OpenSea):} Multi-marketplace sweeping
\item \textbf{Genie (acquired by Uniswap):} NFT shopping cart
\item \textbf{Blur:} Hybrid marketplace/aggregator
\item \textbf{Rarible:} Aggregates OpenSea, LooksRare, X2Y2
\end{itemize}

\vspace{0.3cm}
\textbf{Benefits:}
\begin{itemize}
\item Best price discovery (compare across platforms)
\item Batch buying (sweep floor listings efficiently)
\item Single interface for multiple marketplaces
\end{itemize}

\vspace{0.3cm}
\textbf{Impact:}
\begin{itemize}
\item Reduces marketplace lock-in (easier to switch)
\item Pressures marketplaces to lower fees (commoditization)
\end{itemize}
\end{frame}

\begin{frame}{Floor Price Dynamics}
\textbf{Floor Price:} Lowest listed price for any NFT in a collection

\vspace{0.3cm}
\textbf{Why Floor Price Matters:}
\begin{itemize}
\item Proxy for collection value and market sentiment
\item Entry point for new buyers (``buying the floor'')
\item Psychological support level (panic selling if broken)
\end{itemize}

\vspace{0.3cm}
\textbf{Floor Price Manipulation:}
\begin{itemize}
\item \textbf{Artificial support:} Whales buy floor listings to prevent drops
\item \textbf{Floor sweeping:} Coordinated buying to create FOMO
\item \textbf{Fake listings:} List at inflated prices to manipulate perception
\end{itemize}

\vspace{0.3cm}
\textbf{Healthier Metrics:}
\begin{itemize}
\item Median price (less susceptible to outliers)
\item 7-day sales volume (actual liquidity)
\item Unique holder count (distribution health)
\end{itemize}
\end{frame}

\begin{frame}{Marketplace Security and Scams}
\textbf{Common NFT Marketplace Scams:}

\vspace{0.3cm}
\begin{enumerate}
\item \textbf{Phishing sites:} Fake marketplace clones steal wallet signatures
\item \textbf{Malicious approvals:} Trick users into approving scam contracts
\item \textbf{Fake collections:} Counterfeit versions of popular NFTs
\item \textbf{Bid manipulation:} Bait-and-switch (cancel bid after acceptance)
\item \textbf{Airdrop scams:} Fake NFTs sent to wallets with malicious links
\end{enumerate}

\vspace{0.3cm}
\textbf{Protection Measures:}
\begin{itemize}
\item Verify marketplace URL (bookmark official sites)
\item Check collection verification badge
\item Review smart contract permissions before signing
\item Use hardware wallets for high-value NFTs
\item Revoke unused approvals (revoke.cash)
\end{itemize}
\end{frame}

\begin{frame}{Cross-Chain NFT Trading}
\textbf{Multi-Chain Expansion:}

\vspace{0.3cm}
\begin{itemize}
\item \textbf{Ethereum:} Dominant NFT ecosystem (70\%+ market share)
\item \textbf{Solana:} Low fees, fast transactions (Magic Eden)
\item \textbf{Polygon:} Ethereum sidechain (gasless minting on OpenSea)
\item \textbf{Arbitrum/Optimism:} Layer 2 scaling (lower gas costs)
\item \textbf{Base:} Coinbase L2 (emerging NFT activity)
\end{itemize}

\vspace{0.3cm}
\textbf{Challenges:}
\begin{itemize}
\item Fragmented liquidity across chains
\item Bridge risks (cross-chain NFT transfers)
\item Ecosystem lock-in (wallets, tools, communities)
\end{itemize}

\vspace{0.3cm}
\textbf{Cross-Chain Bridges:}
\begin{itemize}
\item Wormhole, LayerZero enable NFT transfers between chains
\item Risk: Smart contract vulnerabilities in bridges
\end{itemize}
\end{frame}

\begin{frame}{Marketplace Revenue Models}
\textbf{Historical Model (2017-2022):}
\begin{itemize}
\item Platform takes 2-5\% fee on all sales

\item OpenSea revenue peak: \$300M+ (2021)
\end{itemize}

\textbf{Zero-Fee Disruption (2022+):}
\begin{itemize}

\vspace{0.3cm}
\textbf{Alternative Revenue Streams:}
\begin{itemize}
\item Token appreciation (BLUR, RARI)

\item Partnerships and integrations
\end{itemize}
\end{frame}

\begin{frame}{OpenSea Seaport: Open Protocol}
\textbf{Seaport (2022):} Open-source NFT marketplace protocol

\vspace{0.3cm}
\textbf{Key Features:}
\begin{itemize}
\item \textbf{Permissionless:} Anyone can build marketplace frontends
\item \textbf{Advanced order types:} Criteria-based, partial fills, bundle trades
\item \textbf{Gas optimization:} Efficient settlement mechanisms
\item \textbf{No protocol fees:} Frontends set their own fees
\end{itemize}

\vspace{0.3cm}
\textbf{Strategic Intent:}
\begin{itemize}
\item Commoditize marketplace infrastructure
\item Compete on UX and features (not protocol lock-in)
\item Foster ecosystem of Seaport-powered marketplaces
\end{itemize}

\vspace{0.3cm}
\textbf{Adoption:} Coinbase NFT, GameStop NFT use Seaport protocol
\end{frame}

\begin{frame}{Marketplace Aggregation and Liquidity}
\textbf{Liquidity Fragmentation:}
\begin{itemize}
\item Same NFT collection listed on multiple marketplaces
\item Buyers must check all platforms for best price
\item Inefficient price discovery
\end{itemize}

\vspace{0.3cm}
\textbf{Aggregators Solve:}
\begin{itemize}
\item Unified liquidity view (all listings in one interface)
\item Automated best price routing
\item Batch transactions (sweep across marketplaces)
\end{itemize}

\vspace{0.3cm}
\textbf{Result:}
\begin{itemize}
\item Marketplaces compete on features, not liquidity lock-in
\item Downward pressure on fees (commoditization)
\item Better experience for traders (price efficiency)
\end{itemize}
\end{frame}

\begin{frame}{Key Takeaways}
\begin{enumerate}
\item NFT marketplaces compete on fees, features, and trader incentives (OpenSea, Blur, Rarible)
\item Listing mechanics include fixed price, auctions, and off-chain order matching (Seaport)
\item Marketplace fees have dropped to 0-2.5\%, creator royalties now optional (royalty debate)
\item Wash trading inflates volume for rewards and visibility (detection via wallet analysis)
\item Aggregators improve liquidity and price discovery (Gem, Genie, Blur)
\item Zero-fee models raise sustainability questions (alternative revenue needed)
\end{enumerate}
\end{frame}
