\begin{frame}{Exercise 1: Browse Collections}
\textbf{Navigate OpenSea:}
\begin{enumerate}
    \item Visit: \url{https://opensea.io}
    \item Explore Rankings: Top collections by volume
    \item Select a collection (e.g., Bored Apes, Azuki, Pudgy Penguins)
    \item Note: floor price, total volume, owner count
\end{enumerate}

\vspace{0.3cm}
\textbf{Collection Stats to Record:}
\begin{itemize}
    \item Total items and unique owners
    \item Floor price and 7-day volume
    \item Creator royalty percentage
    \item Contract verification status
\end{itemize}
\end{frame}

\begin{frame}{Exercise 2: Metric Analysis}
\textbf{Key Metrics Explained:}
\begin{itemize}
    \item \textbf{Floor Price}: Lowest listed price
    \item \textbf{Volume}: Total trading activity (ETH)
    \item \textbf{Listed Ratio}: \% of collection for sale
    \item \textbf{Unique Owners}: Distribution measure
\end{itemize}

\vspace{0.3cm}
\textbf{Analysis Questions:}
\begin{enumerate}
    \item Is floor price stable or volatile?
    \item High volume with low owners = wash trading?
    \item What \% of collection is listed? (High = bearish)
\end{enumerate}
\end{frame}

\begin{frame}{Exercise 3: Rarity Analysis}
\textbf{Tools:}
\begin{itemize}
    \item Rarity.tools
    \item Trait Sniper
    \item OpenSea trait filters
\end{itemize}

\vspace{0.3cm}
\textbf{Tasks:}
\begin{enumerate}
    \item Select an item from your chosen collection
    \item List all traits and their rarity percentages
    \item Calculate approximate rarity rank
    \item Compare price to similar rarity items
\end{enumerate}
\end{frame}

\begin{frame}{Key Takeaways}
\begin{itemize}
    \item Floor price is the key metric for collection health
    \item High unique owner count indicates strong distribution
    \item Rarity drives individual NFT pricing
    \item Watch for wash trading (high volume, few owners)
    \item Verify contract and creator reputation
\end{itemize}
\end{frame}
