\begin{frame}{Activity 1: Collection Discovery}
\textbf{Task:} Find and evaluate trending NFT collections

\vspace{0.3cm}
\textbf{Steps:}
\begin{enumerate}
\item Navigate to OpenSea homepage (\url{https://opensea.io})
\item Browse ``Trending'' collections (24h, 7d, 30d volume)
\item Filter by category (Art, Collectibles, Gaming, etc.)
\item Select a collection with >1 ETH floor price and >500 items
\end{enumerate}

\vspace{0.3cm}
\textbf{Key Metrics to Record:}
\begin{itemize}
\item Floor price (ETH)
\item 24h/7d volume (ETH)
\item Total supply (number of items)
\item Unique owners (holder distribution)
\item Total sales and average sale price
\end{itemize}
\end{frame}

\begin{frame}{OpenSea Collection Page Anatomy}
\textbf{Top Section:}
\begin{itemize}
\item Collection banner and profile image
\item Verification badge (blue checkmark)
\item Social links (Twitter, Discord, website)
\end{itemize}

\vspace{0.3cm}
\textbf{Key Statistics:}
\begin{itemize}

\end{frame}

\begin{frame}[fragile]{Activity 2: Metadata Inspection}
\textbf{Task:} Extract and decode NFT metadata from IPFS

\vspace{0.3cm}
\textbf{Steps:}
\begin{enumerate}
\item Select a specific NFT from the collection
\item Click ``Details'' section to expand
\item Find ``Contract Address'' and ``Token ID''
\item Navigate to Etherscan contract page
\item Call \texttt{tokenURI(tokenId)} function in ``Read Contract'' tab
\item Copy the returned URI (IPFS or HTTP)
\item Access metadata via IPFS gateway (if IPFS URI)
\end{enumerate}

\vspace{0.3cm}
\textbf{Example IPFS URI:}
\begin{verbatim}
ipfs://QmXyZ123.../1234.json
\end{verbatim}

\textbf{Convert to Gateway URL:}
\begin{verbatim}
https://ipfs.io/ipfs/QmXyZ123.../1234.json
\end{verbatim}
\end{frame}

\begin{frame}[fragile]{Decoding Metadata JSON}
\textbf{Typical Metadata Structure:}

\vspace{0.3cm}
\begin{verbatim}
{
  "name": "Cool NFT #1234",
  "description": "A unique digital collectible",
  "image": "ipfs://QmAbc.../1234.png",
  "attributes": [
    {"trait_type": "Background", "value": "Blue"},
    {"trait_type": "Eyes", "value": "Laser"},
    {"trait_type": "Hat", "value": "Crown"}
  ]
}
\end{verbatim}

\vspace{0.3cm}
\textbf{Analysis Questions:}
\begin{itemize}
\item Is the image stored on IPFS or centralized server?
\item How many attributes does the NFT have?
\item Are trait values descriptive and meaningful?
\end{itemize}
\end{frame}

\begin{frame}{Activity 3: Floor Price Tracking}
\textbf{Task:} Analyze floor price trends and volatility

\vspace{0.3cm}
\textbf{Steps:}
\begin{enumerate}
\item Navigate to collection ``Activity'' chart
\item Switch to ``Floor Price'' view (7d or 30d)
\item Record daily floor price (create spreadsheet)
\item Calculate daily percentage change
\item Identify support and resistance levels
\end{enumerate}

\vspace{0.3cm}
\textbf{Analysis Metrics:}
\begin{itemize}
\item \textbf{Volatility:} Standard deviation of daily changes
\item \textbf{Trend:} Upward, downward, or sideways movement
\item \textbf{Support:} Price level with repeated buying interest
\item \textbf{Correlation:} Does floor move with ETH price?
\end{itemize}

\vspace{0.3cm}
\textbf{Red Flag:} Steep, continuous decline without stabilization
\end{frame}

\begin{frame}{Activity 4: Volume and Liquidity Analysis}
\textbf{Task:} Evaluate trading activity and market depth

\vspace{0.3cm}
\textbf{Key Questions:}
\begin{enumerate}
\item What is the 7-day trading volume (ETH)?
\item How many unique buyers traded in the last 7 days?
\item What percentage of the collection is currently listed?
\item Are there collection-wide offers (bids)?
\end{enumerate}

\vspace{0.3cm}
\textbf{Liquidity Indicators:}
\begin{itemize}
\item \textbf{High liquidity:} >10\% listed, daily sales, tight bid-ask spread
\item \textbf{Medium liquidity:} 5-10\% listed, weekly sales
\item \textbf{Low liquidity:} <5\% listed, infrequent sales, wide bid-ask
\end{itemize}

\vspace{0.3cm}
\textbf{Warning:} Low liquidity = difficulty selling at floor price
\end{frame}

\begin{frame}{Activity 5: Holder Distribution}
\textbf{Task:} Analyze ownership concentration

\vspace{0.3cm}
\textbf{Steps:}
\begin{enumerate}
\item Note total supply and unique owners
\item Calculate ownership ratio: Unique Owners / Total Supply
\item Use Etherscan or Nansen to identify top holders
\item Check if team/founders hold significant percentage
\end{enumerate}

\vspace{0.3cm}
\textbf{Healthy Distribution:}
\begin{itemize}
\item Ownership ratio >60\% (wide distribution)
\item Top 10 holders own <20\% of supply
\item No single whale controlling >10\%
\end{itemize}

\vspace{0.3cm}
\textbf{Red Flags:}
\begin{itemize}
\item Ownership ratio <40\% (concentrated holdings)
\item Team holds >30\% (dump risk)
\item Single wallet owns >15\% (manipulation risk)
\end{itemize}
\end{frame}

\begin{frame}{Activity 6: Rarity Analysis}
\textbf{Task:} Use trait filters to identify rare NFTs

\vspace{0.3cm}
\textbf{Steps:}
\begin{enumerate}
\item Click ``Filter'' on OpenSea collection page
\item Expand trait categories (Background, Eyes, Hat, etc.)
\item Note the count for each trait value (frequency)
\item Calculate rarity: (Trait count / Total supply) * 100\%
\item Identify traits with <5\% frequency (rare)
\end{enumerate}

\vspace{0.3cm}
\textbf{Example Calculation:}
\begin{itemize}
\item Collection size: 10,000 NFTs
\item Laser Eyes trait: 50 NFTs have this trait
\item Rarity: (50 / 10,000) * 100 = 0.5\% (very rare)
\end{itemize}

\vspace{0.3cm}
\textbf{Rarity Premium:} Rare traits typically command 2-10x floor price
\end{frame}

\begin{frame}{Rarity Tools and Scoring}
\textbf{Rarity Scoring Methods:}

\vspace{0.3cm}
\begin{enumerate}
\item \textbf{Trait Rarity:} Sum of individual trait rarities
\item \textbf{Statistical Rarity:} Product of trait probabilities
\item \textbf{Trait Count:} NFTs with more traits = rarer
\end{enumerate}

\vspace{0.3cm}
\textbf{External Rarity Tools:}
\begin{itemize}
\item \textbf{rarity.tools:} Comprehensive rarity rankings
\item \textbf{Rarity Sniper:} Real-time rarity scores
\item \textbf{Traitsniper:} Trait-based filtering and scoring
\end{itemize}

\vspace{0.3cm}
\textbf{Using rarity.tools:}
\begin{enumerate}
\item Search for collection name
\item View ranked list (Rank \#1 = rarest)
\item Compare OpenSea listings to rarity ranks
\item Identify underpriced rare NFTs (arbitrage opportunity)
\end{enumerate}
\end{frame}

\begin{frame}{Activity 7: Smart Contract Verification}
\textbf{Task:} Verify collection authenticity and security

\vspace{0.3cm}
\textbf{Steps:}
\begin{enumerate}
\item Copy contract address from OpenSea ``Details''
\item Navigate to Etherscan: \url{https://etherscan.io}
\item Paste contract address in search bar
\item Check for verified contract (green checkmark)
\item Review contract source code (if verified)
\item Confirm it uses standard ERC-721 (OpenZeppelin)
\end{enumerate}

\vspace{0.3cm}
\textbf{Red Flags:}
\begin{itemize}
\item Contract not verified (cannot audit code)
\item Custom implementation (higher risk)
\item Ownership not renounced (team can modify)
\item Recent contract deployment (<1 month old)
\end{itemize}
\end{frame}

\begin{frame}{Activity 8: Community and Social Signals}
\textbf{Task:} Evaluate project community strength

\vspace{0.3cm}
\textbf{Metrics to Check:}
\begin{itemize}
\item \textbf{Twitter:} Follower count, engagement rate, activity
\item \textbf{Discord:} Member count, daily active users, moderation
\item \textbf{Website:} Professional design, roadmap, team info
\item \textbf{OpenSea:} Verified badge (blue checkmark)
\end{itemize}

\vspace{0.3cm}
\textbf{Healthy Community:}
\begin{itemize}
\item Twitter: >10k followers, regular updates
\item Discord: >5k members, active daily chat
\item Transparent team (doxxed or credible pseudonymous)
\end{itemize}

\vspace{0.3cm}
\textbf{Warning Signs:}
\begin{itemize}
\item Bot followers (low engagement despite high count)
\item Inactive social channels (last post >1 week ago)
\item Anonymous team with no track record
\end{itemize}
\end{frame}

\begin{frame}{Investment Evaluation Framework}
\textbf{Due Diligence Checklist:}

\vspace{0.3cm}
\begin{enumerate}
\item \textbf{Contract Security:} Verified, standard ERC-721, ownership renounced
\item \textbf{Metadata Storage:} IPFS or Arweave (not HTTP), immutable URI
\item \textbf{Holder Distribution:} >60\% ownership ratio, no whale dominance
\item \textbf{Liquidity:} >5\% listed, daily sales, reasonable bid-ask spread
\item \textbf{Community:} Active social channels, transparent team, engaged holders
\item \textbf{Floor Price Stability:} Support levels, no steep decline
\item \textbf{Rarity Structure:} Clear trait hierarchy, rarity premium exists
\item \textbf{Utility:} Beyond speculation (IP rights, ecosystem access, etc.)
\end{enumerate}

\vspace{0.3cm}
\textbf{Scoring:} 1 point per criterion met, 6+ = Acceptable risk, 8+ = Strong project
\end{frame}

\begin{frame}{Red Flags: Avoid These Collections}
\textbf{Critical Warning Signs:}

\vspace{0.3cm}
\begin{enumerate}
\item \textbf{No verified contract:} Cannot audit code (scam risk)
\item \textbf{HTTP metadata:} Centralized server can disappear
\item \textbf{Mutable tokenURI:} Team can change metadata (rug risk)
\item \textbf{Team holds >50\%:} Dump risk, manipulation
\item \textbf{Zero liquidity:} No sales in 7+ days, <2\% listed
\item \textbf{Inactive community:} No social activity, dead Discord
\item \textbf{Copycat project:} Obvious clone of popular collection
\item \textbf{Anonymous team, no roadmap:} Likely pump-and-dump
\end{enumerate}

\vspace{0.3cm}
\textbf{Rule:} Any 2+ critical red flags = Avoid investment
\end{frame}
