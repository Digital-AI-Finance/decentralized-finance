\section{Module Recaps}

\begin{frame}{Module A: Foundations -- Key Takeaways}
\begin{itemize}
\item \textbf{Blockchain = Distributed Ledger}: Immutable, transparent, decentralized
\item \textbf{Cryptography}: Hashing (SHA-256), digital signatures (ECDSA), Merkle trees
\item \textbf{Bitcoin Invention}: Satoshi Nakamoto's solution to double-spending
\item \textbf{PoW Consensus}: Miners compete to solve hash puzzle, longest chain wins
\item \textbf{PoS Evolution}: Energy-efficient, validators stake tokens
\item \textbf{Byzantine Fault Tolerance}: 2/3 honest nodes required
\vspace{0.3cm}
\item \textbf{Fundamental Insight}: Trust through math and game theory, not institutions
\end{itemize}
\end{frame}

\begin{frame}{Module B: Cryptocurrencies -- Key Takeaways}
\begin{itemize}
\item \textbf{Bitcoin}: Digital gold, 21M supply cap, halving every 4 years
\item \textbf{UTXO Model}: Transaction inputs/outputs, not account balances
\item \textbf{Mining}: Hashrate competition, ASIC dominance, mining pools
\item \textbf{Wallets}: Hot (online, convenient) vs cold (offline, secure)
\item \textbf{Altcoins}: Litecoin (faster), Monero (privacy), Bitcoin Cash (bigger blocks)
\item \textbf{Network Effects}: Bitcoin's first-mover advantage and brand recognition
\vspace{0.3cm}
\item \textbf{Critical Lesson}: Private key = ownership; lose key = lose funds
\end{itemize}
\end{frame}

\begin{frame}{Module C: Ethereum \& Smart Contracts -- Key Takeaways}
\begin{itemize}
\item \textbf{Smart Contracts}: Self-executing code, unstoppable applications
\item \textbf{Solidity}: Most popular smart contract language (JavaScript-like)
\item \textbf{EVM}: Turing-complete virtual machine, gas mechanism prevents infinite loops
\item \textbf{dApps}: Frontend + smart contract backend (MetaMask integration)
\item \textbf{The Merge (2022)}: Ethereum switched from PoW to PoS (99\% energy reduction)
\item \textbf{ERC Standards}: ERC-20 (tokens), ERC-721 (NFTs), ERC-1155 (multi-token)
\vspace{0.3cm}
\item \textbf{Paradigm Shift}: From ``code is code'' to ``code is law''
\end{itemize}
\end{frame}

\begin{frame}{Module D: DeFi -- Key Takeaways}
\begin{itemize}
\item \textbf{DeFi = Open Financial System}: No intermediaries, permissionless
\item \textbf{AMMs (Uniswap)}: x * y = k, liquidity pools replace order books
\item \textbf{Lending (Aave, Compound)}: Over-collateralized loans, algorithmic interest rates
\item \textbf{Stablecoins}: USDC (fiat-backed), DAI (crypto-collateralized), UST (algorithmic, failed)
\item \textbf{Yield Farming}: Liquidity mining, APY chasing, impermanent loss risk
\item \textbf{Derivatives}: Perpetual futures (funding rates), options (Deribit, Opyn)
\item \textbf{TVL}: \$100B+ locked in DeFi protocols (peak 2021, recovered 2024)
\vspace{0.3cm}
\item \textbf{Innovation vs Risk}: Composability enables innovation, also cascading failures
\end{itemize}
\end{frame}

\begin{frame}{Module E: NFTs \& Tokenization -- Key Takeaways}
\begin{itemize}
\item \textbf{NFTs}: Unique tokens, provable ownership, on-chain metadata
\item \textbf{Use Cases}: Digital art, gaming assets, memberships, ticketing
\item \textbf{Marketplaces}: OpenSea, Blur, Magic Eden (Solana)
\item \textbf{Royalties}: Creators earn on secondary sales (on-chain enforcement)
\item \textbf{Tokenomics}: Token design, utility vs governance, vesting schedules
\item \textbf{DAOs}: On-chain governance, token-weighted voting, treasury management
\item \textbf{2021-2022 Boom/Bust}: \$25B NFT market peak, 90\%+ correction, niche survival
\vspace{0.3cm}
\item \textbf{Lasting Impact}: Digital ownership infrastructure, beyond speculative JPEGs
\end{itemize}
\end{frame}

\begin{frame}{Module F: Advanced Topics -- Key Takeaways}
\begin{itemize}
\item \textbf{Layer 2 Scaling}:
\begin{itemize}
\item Optimistic Rollups (7-day withdrawal, fraud proofs)
\item ZK-Rollups (instant finality, validity proofs)
\item State channels (Lightning Network for Bitcoin)
\end{itemize}
\item \textbf{Flash Loans}: Uncollateralized, atomic transactions (arbitrage, attacks)
\item \textbf{Composability}: DeFi as ``money legos'' -- powerful but risky
\item \textbf{Smart Contract Security}:
\begin{itemize}
\item Reentrancy (The DAO hack), oracle manipulation, access control
\item Tools: Slither, Mythril, formal verification
\item Defense in depth: Audits + bug bounties + monitoring
\end{itemize}
\vspace{0.3cm}
\item \textbf{Critical Lesson}: \$3B+ lost to exploits; security is paramount
\end{itemize}
\end{frame}

\begin{frame}{Module G: Regulation \& Future -- Key Takeaways}
\begin{itemize}
\item \textbf{Regulatory Spectrum}: Hostile (China ban) to permissive (Switzerland, Singapore)
\item \textbf{US}: Fragmented (SEC vs CFTC), regulation by enforcement
\item \textbf{EU MiCA}: Comprehensive framework, stablecoin focus, CASP licensing
\item \textbf{Switzerland}: Principles-based, clear token classification, DLT Act
\item \textbf{CBDCs}: 130+ countries exploring, e-CNY largest pilot
\item \textbf{Future Trends}:
\begin{itemize}
\item Institutional adoption, RWA tokenization, AI+crypto convergence
\item Account abstraction (UX breakthrough), modular blockchains
\item ZK proofs (privacy + scalability), DePIN (decentralized infrastructure)
\end{itemize}
\vspace{0.3cm}
\item \textbf{Tension}: Innovation vs regulation, privacy vs compliance
\end{itemize}
\end{frame}

\section{Cross-Cutting Themes}

\begin{frame}{Theme 1: Decentralization vs Efficiency Tradeoff}
\begin{itemize}
\item \textbf{Blockchain Trilemma}: Decentralization, security, scalability (pick 2)
\item \textbf{Bitcoin}: Maximally decentralized, sacrifices scalability (7 TPS)
\item \textbf{Ethereum}: Decentralized, adding scalability via Layer 2
\item \textbf{Solana}: High throughput (65,000 TPS), less decentralized (higher hardware requirements)
\item \textbf{Centralized Exchanges}: Maximum efficiency, zero decentralization (FTX collapse risk)
\item \textbf{Layer 2 Solutions}: Pragmatic compromise (rollups inherit L1 security)
\vspace{0.3cm}
\item \textbf{Design Philosophy}: No perfect solution, context determines optimal tradeoff
\end{itemize}
\end{frame}

\begin{frame}{Theme 2: Code is Law -- Until It Isn't}
\begin{itemize}
\item \textbf{Smart Contract Immutability}: Code executes exactly as written
\item \textbf{The DAO Hack (2016)}: \$60M drained via reentrancy
\begin{itemize}
\item Community decision: Hard fork to reverse (Ethereum vs Ethereum Classic split)
\item ``Code is law'' vs ``code has bugs, community decides''
\end{itemize}
\item \textbf{Tornado Cash Sanctions (2022)}: US Treasury sanctioned smart contract
\begin{itemize}
\item Debate: Can you sanction code? Is code speech or conduct?
\end{itemize}
\item \textbf{Legal Enforcement}: SBF convicted despite decentralization rhetoric
\vspace{0.3cm}
\item \textbf{Reality}: Code operates within legal and social contexts
\item \textbf{Governance}: Most protocols have upgrade mechanisms (not truly immutable)
\end{itemize}
\end{frame}

\begin{frame}{Theme 3: Speculation vs Utility}
\begin{itemize}
\item \textbf{Speculation Dominates} (2024 reality):
\begin{itemize}
\item 90\%+ crypto transactions are trading/speculation
\item Memecoins, pump-and-dump schemes, retail FOMO
\item Price volatility hinders adoption as currency
\end{itemize}
\item \textbf{Utility Emerging}:
\begin{itemize}
\item Stablecoins: \$150B market cap, actual payment usage (remittances)
\item NFT ticketing: Ticketmaster, sports teams adopting
\item DeFi primitives: Real financial services (lending, derivatives)
\item Supply chain: Walmart, IBM Food Trust (provenance tracking)
\end{itemize}
\item \textbf{Adoption S-Curve}:
\begin{itemize}
\item Early adopters (2010-2020): Speculation, ideological
\item Crossing the chasm (2024-2030): Institutional, utility-driven
\end{itemize}
\vspace{0.3cm}
\item \textbf{Long-term:} Utility will dominate, speculation will remain (like stock market)
\end{itemize}
\end{frame}

\begin{frame}{Theme 4: Permissionless Innovation}
\begin{itemize}
\item \textbf{Traditional Finance}: Permission required (licenses, capital, compliance)
\item \textbf{DeFi:} Deploy smart contract, anyone can use (no asking permission)
\item \textbf{Innovation Explosion:}
\begin{itemize}
\item Uniswap: 2 developers, \$100B+ volume
\item Compound: Automated interest rates, no credit committee
\item Aave: Flash loans (impossible in TradFi)
\end{itemize}
\item \textbf{Composability:} Build on existing protocols without partnerships
\begin{itemize}
\item Yearn Finance: Aggregates lending protocols
\item 1inch: Aggregates DEXs
\item Curve Wars: Protocols competing for liquidity incentives
\end{itemize}
\item \textbf{Downside:} Exploits, scams, regulatory uncertainty
\vspace{0.3cm}
\item \textbf{Paradigm:} ``Move fast and break things'' applied to finance
\end{itemize}
\end{frame}

\begin{frame}{Theme 5: Centralization Creep}
\begin{itemize}
\item \textbf{Paradox:} Decentralized protocols often have centralized components
\item \textbf{Examples:}
\begin{itemize}
\item \textbf{Exchanges}: Binance, Coinbase dominate (70\%+ trading volume)
\item \textbf{Stablecoins}: Circle (USDC) can freeze accounts
\item \textbf{Infrastructure:} Infura, Alchemy (most dApps use centralized RPC)
\item \textbf{Mining/Staking:} Large pools control majority hashrate/stake
\item \textbf{Development:} Ethereum Foundation, Consensus (ConsenSys) have outsized influence
\end{itemize}
\item \textbf{Nakamoto Coefficient:} Measure of decentralization (min entities to halt network)
\begin{itemize}
\item Bitcoin: ~4 mining pools
\item Ethereum: ~3 entities (Lido, Coinbase, Kraken)
\item Solana: ~19 validators
\end{itemize}
\vspace{0.3cm}
\item \textbf{Challenge:} Decentralization is a spectrum, not binary
\end{itemize}
\end{frame}

\begin{frame}{Theme 6: Privacy vs Transparency}
\begin{itemize}
\item \textbf{Blockchain Transparency:} All transactions public (pseudonymous, not anonymous)
\item \textbf{Chain Analysis:} Chainalysis, Elliptic track illicit funds (law enforcement cooperation)
\item \textbf{Privacy Coins}:
\begin{itemize}
\item Monero: Ring signatures, stealth addresses (truly private)
\item Zcash: Optional privacy (zk-SNARKs)
\item Regulatory pressure: Exchanges delisting (AML concerns)
\end{itemize}
\item \textbf{Privacy Tools:}
\begin{itemize}
\item Tornado Cash: Mixer (sanctioned by US Treasury)
\item Aztec: Privacy on Ethereum (ZK proofs)
\end{itemize}
\item \textbf{Regulatory Conflict:}
\begin{itemize}
\item Financial surveillance (AML/CFT, tax enforcement)
\item Human right to privacy (dissidents, activists)
\end{itemize}
\vspace{0.3cm}
\item \textbf{Future:} ZK proofs enable privacy + compliance (prove legality without revealing transaction)
\end{itemize}
\end{frame}

\section{Critical Reflections}

\begin{frame}{What Blockchain Does Well}
\begin{enumerate}
\item \textbf{Censorship Resistance:} No single point of control, hard to shut down
\item \textbf{Programmable Money:} Smart contracts enable complex financial logic
\item \textbf{Composability:} Protocols integrate permissionlessly
\item \textbf{Global Accessibility:} Internet connection = access (no bank account needed)
\item \textbf{Transparency:} All transactions auditable (reducing corruption)
\item \textbf{24/7 Operation:} No market hours, no weekends
\item \textbf{Rapid Settlement}: Minutes (vs days for traditional finance)
\item \textbf{Collateral Efficiency:} Flash loans, atomic swaps (impossible in TradFi)
\end{enumerate}
\end{frame}

\begin{frame}{What Blockchain Does Poorly}
\begin{enumerate}
\item \textbf{Scalability:} 7-65,000 TPS (vs Visa 65,000 TPS)
\item \textbf{User Experience:} Seed phrases, gas fees, irreversible transactions (no customer support)
\item \textbf{Energy Consumption:} Bitcoin PoW uses ~150 TWh/year (country-level)
\item \textbf{Volatility}: Unsuitable as currency (price swings)
\item \textbf{Regulatory Uncertainty:} Legal status unclear in many jurisdictions
\item \textbf{Fraud/Scams:} Irreversibility enables theft, no recourse
\item \textbf{Complexity:} Steep learning curve (technical barriers)
\item \textbf{Centralization Risk:} Mining pools, staking concentration, infrastructure providers
\end{enumerate}
\end{frame}

\begin{frame}{Legitimate Use Cases (Where Blockchain Adds Value)}
\begin{enumerate}
\item \textbf{Cross-Border Remittances:} Cheaper, faster than Western Union (USDC, Lightning)
\item \textbf{Inflation Hedge}: Store of value in unstable currencies (Argentina, Venezuela)
\item \textbf{Censorship Evasion:} Donations to dissidents (WikiLeaks, Ukraine war donations)
\item \textbf{Financial Inclusion}: Banking for unbanked (Africa, Southeast Asia)
\item \textbf{Tokenization of Assets:} Fractional ownership (real estate, art)
\item \textbf{Supply Chain Provenance:} Tracking goods (diamonds, food safety)
\item \textbf{Decentralized Identity:} Self-sovereign identity (no Facebook/Google control)
\item \textbf{DAOs:} Coordination without centralized hierarchy (GitcoinDAO, Uniswap governance)
\item \textbf{Gaming Economies:} True ownership of in-game assets (Axie Infinity model)
\end{enumerate}
\end{frame}

\begin{frame}{Questionable Use Cases (Blockchain Not Necessary)}
\begin{enumerate}
\item \textbf{Most Supply Chains:} Centralized databases work fine (blockchain overkill)
\item \textbf{Voting:} Security and privacy challenges outweigh benefits
\item \textbf{Medical Records:} Privacy requirements conflict with blockchain transparency
\item \textbf{Most NFTs}: Simple database + digital signature suffices (no need for blockchain)
\item \textbf{Enterprise Blockchains}: Private blockchains = glorified shared databases
\item \textbf{IoT:} High transaction volume, low latency needs (blockchain too slow)
\item \textbf{Most ``Blockchain for X''}: Marketing buzzword, no actual benefit
\vspace{0.3cm}
\item \textbf{Heuristic:} If you can use a database, use a database. Blockchain only when decentralization/censorship-resistance is critical.
\end{enumerate}
\end{frame}

\begin{frame}{The FTX Collapse: Lessons Learned}
\begin{itemize}
\item \textbf{What Happened (November 2022)}:
\begin{itemize}
\item FTX (2nd largest exchange) commingled customer funds with Alameda Research (sister hedge fund)
\item Alameda lost billions in risky bets (Luna collapse, illiquid tokens)
\item Bank run triggered when Binance CEO tweeted concerns
\item \$8B customer funds missing, Sam Bankman-Fried arrested
\end{itemize}
\item \textbf{Lessons:}
\begin{enumerate}
\item \textbf{Not Your Keys, Not Your Coins:} Exchanges are custodians (Mt.Gox, FTX)
\item \textbf{Proof of Reserves Insufficient:} FTX published fake attestations
\item \textbf{Regulation Needed:} Self-regulation failed spectacularly
\item \textbf{Offshore Exchanges Risky:} Light Bahamas oversight enabled fraud
\item \textbf{Due Diligence:} High yields often signal hidden risk
\end{enumerate}
\end{itemize}
\vspace{0.3cm}
\item \textbf{Industry Response:} Exchanges publish Merkle tree proofs, regulatory scrutiny increased
\end{itemize}
\end{frame}

\section{Looking Ahead}

\begin{frame}{2025-2030: Key Questions}
\begin{enumerate}
\item \textbf{Scaling:} Will Layer 2s solve Ethereum's scalability, or will alt-L1s dominate?
\item \textbf{Regulation:} Will global frameworks converge (MiCA template) or fragment?
\item \textbf{CBDCs:} Will they coexist with crypto, or attempt to crowd it out?
\item \textbf{Institutional Adoption:} Will \$1T+ institutional capital enter crypto?
\item \textbf{DeFi vs CeFi:} Which model wins (composable protocols vs regulated intermediaries)?
\item \textbf{Privacy:} Can ZK proofs enable privacy without enabling crime?
\item \textbf{Interoperability:} Will cross-chain bridges remain attack vectors?
\item \textbf{Energy:} Will PoS adoption reduce environmental concerns?
\item \textbf{Quantum Threat:} Will post-quantum cryptography be deployed in time?
\item \textbf{Mainstream UX:} Will account abstraction achieve Web2-level usability?
\end{enumerate}
\end{frame}

\begin{frame}{Optimistic Scenario (2030)}
\begin{itemize}
\item \textbf{Institutional Adoption:} Pension funds, sovereign wealth funds allocate 5-10\% to crypto
\item \textbf{Stablecoins:} \$1T+ market cap, used globally for payments and remittances
\item \textbf{Tokenization:} \$10T+ real-world assets on-chain (real estate, bonds, equities)
\item \textbf{Layer 2 Success:} Ethereum + rollups achieve 100,000+ TPS, fees <\$0.01
\item \textbf{Regulatory Clarity:} Global frameworks converge, DeFi compliance solutions emerge
\item \textbf{Account Abstraction:} Wallets as easy as email (social recovery, gas abstraction)
\item \textbf{Privacy + Compliance:} ZK proofs enable private, compliant transactions
\item \textbf{CBDCs Coexist:} Central bank digital currencies complement, not replace, crypto
\item \textbf{Mainstream Use:} 1B+ users (vs 500M in 2024)
\item \textbf{Impact:} Financial system fundamentally transformed, more inclusive and efficient
\end{itemize}
\end{frame}

\begin{frame}{Pessimistic Scenario (2030)}
\begin{itemize}
\item \textbf{Regulatory Crackdown:} Major jurisdictions ban DeFi, privacy tools, self-custody
\item \textbf{CBDC Monopoly:} Central banks outlaw competing stablecoins, crypto marginalized
\item \textbf{Centralization Wins:} A few mega-platforms (Coinbase, Binance) dominate, surveillance standard
\item \textbf{Scaling Failure:} Layer 2s too complex/fragmented, alt-L1s centralized, fees remain high
\item \textbf{Security Catastrophe:} Major exploit (>\$10B) destroys trust, insurance unavailable
\item \textbf{Quantum Breakthrough:} ECDSA broken, mass theft before migration complete
\item \textbf{Environmental Backlash:} PoW banned globally, PoS attacked as plutocracy
\item \textbf{Stablecoin Collapse:} Tether/Circle fail, contagion wipes out DeFi
\item \textbf{User Apathy:} Complexity, scams, volatility drive users back to TradFi
\item \textbf{Niche Survival:} Crypto becomes niche tool (like Tor), not mainstream finance
\end{itemize}
\end{frame}

\begin{frame}{Realistic Scenario (2030)}
\begin{itemize}
\item \textbf{Hybrid System:} Regulated CeFi + permissionless DeFi coexist
\item \textbf{Institutional Participation:} Via compliant on/off ramps, tokenized securities
\item \textbf{Stablecoins:} Regulated (MiCA-style), dominate crypto payments (\$500B market cap)
\item \textbf{Layer 2 Maturity}: Optimistic + ZK rollups achieve 50,000 TPS, fees \$0.10
\item \textbf{Selective Regulation:} Retail-facing services regulated (exchanges, wallets), protocols mostly exempt
\item \textbf{Privacy Constrained:} KYC for fiat on/off ramps, some privacy tools banned
\item \textbf{CBDCs Launched:} But limited adoption (prefer private stablecoins/crypto)
\item \textbf{Use Cases Validated:} Remittances, tokenization, DAOs mainstream; speculation remains large
\item \textbf{Users:} 750M (50\% growth from 2024), still <10\% global population
\item \textbf{Conclusion:} Crypto as parallel financial system, not replacement
\end{itemize}
\end{frame}

\begin{frame}{Final Thoughts: What Did We Learn?}
\begin{enumerate}
\item \textbf{Technology Enables, Society Decides:} Blockchain is a tool, outcomes depend on adoption and regulation
\item \textbf{Decentralization is Hard:} Tradeoffs everywhere (speed, cost, security, usability)
\item \textbf{Finance is Being Rebuilt:} From first principles, in public, with open source
\item \textbf{Speculation Funds Innovation:} Bubbles are destructive but also fund R\&D (dot-com parallel)
\item \textbf{Security is Existential:} One exploit can destroy years of progress
\item \textbf{Regulation is Inevitable:} Question is whether it enables or stifles innovation
\item \textbf{Privacy Matters:} But conflicts with AML/tax enforcement (unresolved tension)
\item \textbf{User Experience:} Crypto won't achieve mainstream adoption until UX matches Web2
\item \textbf{The Future is Uncertain:} Could be revolutionary or a niche curiosity
\end{enumerate}
\end{frame}

\begin{frame}{Critical Thinking: Question Everything}
\begin{itemize}
\item \textbf{Don't Trust, Verify:} Crypto ethos applies to crypto itself
\begin{itemize}
\item Verify token contracts before investing (scams everywhere)
\item Audit smart contracts before depositing funds
\item Check proof of reserves (exchanges, stablecoins)
\end{itemize}
\item \textbf{Beware Hype Cycles}:
\begin{itemize}
\item 2017: ICO mania (95\%+ projects failed)
\item 2021: NFT boom (90\%+ floor price collapse)
\item 2024: AI + crypto hype (TBD)
\end{itemize}
\item \textbf{Cui Bono (Who Benefits)?}:
\begin{itemize}
\item Token launches: Founders, VCs dump on retail
\item Yield farming: Early LPs extract value from late entrants
\item Influencer shills: Paid promotions disguised as education
\end{itemize}
\item \textbf{Intellectual Humility}: This field evolves rapidly, today's truth may be tomorrow's outdated belief
\end{itemize}
\end{frame}
