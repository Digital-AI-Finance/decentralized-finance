\section{Module Recaps}

\begin{frame}{Module A: Foundations -- Key Takeaways}
\begin{itemize}
\item \textbf{Blockchain = Distributed Ledger}: Immutable, transparent, decentralized
\item \textbf{Cryptography}: Hashing (SHA-256), digital signatures (ECDSA), Merkle trees
\item \textbf{Bitcoin Invention}: Satoshi Nakamoto's solution to double-spending
\item \textbf{PoW Consensus}: Miners compete to solve hash puzzle, longest chain wins
\item \textbf{PoS Evolution}: Energy-efficient, validators stake tokens
\item \textbf{Byzantine Fault Tolerance}: 2/3 honest nodes required
\vspace{0.3cm}
\item \textbf{Fundamental Insight}: Trust through math and game theory, not institutions
\end{itemize}
\end{frame}

\begin{frame}{Module B: Ethereum -- Key Takeaways}
\begin{itemize}
\item \textbf{Smart Contracts}: Self-executing code, unstoppable applications
\item \textbf{Solidity}: Most popular smart contract language (JavaScript-like)
\item \textbf{EVM}: Turing-complete virtual machine, gas mechanism prevents infinite loops
\item \textbf{dApps}: Frontend + smart contract backend (MetaMask integration)
\item \textbf{The Merge (2022)}: Ethereum switched from PoW to PoS (99\% energy reduction)
\item \textbf{ERC Standards}: ERC-20 (tokens), ERC-721 (NFTs), ERC-1155 (multi-token)
\vspace{0.3cm}
\item \textbf{Paradigm Shift}: From ``code is code'' to ``code is law''
\end{itemize}
\end{frame}

\begin{frame}{Theme 4: Permissionless Innovation}
\begin{itemize}
\item \textbf{Traditional Finance}: Permission required (licenses, capital, compliance)
\item \textbf{DeFi:} Deploy smart contract, anyone can use (no asking permission)
\item \textbf{Innovation Explosion:}
\begin{itemize}
\item Uniswap: 2 developers, \$100B+ volume
\item Compound: Automated interest rates
\item Aave: Flash loans (impossible in TradFi)
\end{itemize}
\item \textbf{Composability:} Build on existing protocols without partnerships
\item \textbf{Downside:} Exploits, scams, regulatory uncertainty
\vspace{0.3cm}
\item \textbf{Paradigm:} ``Move fast and break things'' applied to finance
\end{itemize}
\end{frame}

\begin{frame}{Theme 5: Centralization Creep}
\begin{itemize}
\item \textbf{Paradox:} Decentralized protocols often have centralized components
\item \textbf{Examples:}
\begin{itemize}
\item \textbf{Exchanges}: Binance, Coinbase dominate (70\%+ trading volume)
\item \textbf{Stablecoins}: Circle (USDC) can freeze accounts
\item \textbf{Infrastructure:} Infura, Alchemy (most dApps use centralized RPC)
\item \textbf{Staking:} Lido controls 30\%+ of staked ETH
\end{itemize}
\item \textbf{Nakamoto Coefficient:} Measure of decentralization
\begin{itemize}
\item Bitcoin: ~4 mining pools
\item Ethereum: ~3 entities (Lido, Coinbase, Kraken)
\end{itemize}
\vspace{0.3cm}
\item \textbf{Challenge:} Decentralization is a spectrum, not binary
\end{itemize}
\end{frame}

\section{Critical Reflections}

\begin{frame}{What Blockchain Does Well}
\begin{enumerate}
\item \textbf{Censorship Resistance:} No single point of control, hard to shut down
\item \textbf{Programmable Money:} Smart contracts enable complex financial logic
\item \textbf{Composability:} Protocols integrate permissionlessly
\item \textbf{Global Accessibility:} Internet connection = access (no bank needed)
\item \textbf{Transparency:} All transactions auditable
\item \textbf{24/7 Operation:} No market hours, no weekends
\item \textbf{Rapid Settlement}: Minutes (vs days for traditional finance)
\end{enumerate}
\end{frame}

\begin{frame}{What Blockchain Does Poorly}
\begin{enumerate}
\item \textbf{Scalability:} 7-65,000 TPS (vs Visa 65,000 TPS)
\item \textbf{User Experience:} Seed phrases, gas fees, irreversible transactions
\item \textbf{Energy Consumption:} Bitcoin PoW uses ~150 TWh/year
\item \textbf{Volatility}: Unsuitable as currency (price swings)
\item \textbf{Regulatory Uncertainty:} Legal status unclear in many jurisdictions
\item \textbf{Fraud/Scams:} Irreversibility enables theft, no recourse
\item \textbf{Complexity:} Steep learning curve (technical barriers)
\end{enumerate}
\end{frame}

\begin{frame}{Legitimate vs Questionable Use Cases}
\begin{columns}[T]
\begin{column}{0.48\textwidth}
\textbf{Blockchain Adds Value}
\begin{itemize}
\item Cross-border remittances
\item Inflation hedge (Argentina, Venezuela)
\item Censorship evasion (dissidents)
\item Financial inclusion (unbanked)
\item Asset tokenization
\item Supply chain provenance
\item Decentralized identity
\end{itemize}
\end{column}
\begin{column}{0.48\textwidth}
\textbf{Blockchain Not Necessary}
\begin{itemize}
\item Most supply chains
\item Voting systems
\item Medical records
\item Most NFTs (simple DB suffices)
\item Enterprise blockchains
\item IoT (too slow)
\item Most ``Blockchain for X''
\end{itemize}
\vspace{0.2cm}
\textbf{Heuristic}: If a database works, use it
\end{column}
\end{columns}
\end{frame}

\section{Looking Ahead}

\begin{frame}{2025-2030: Key Questions}
\begin{enumerate}
\item \textbf{Scaling:} Will Layer 2s solve Ethereum's scalability, or will alt-L1s dominate?
\item \textbf{Regulation:} Will global frameworks converge (MiCA template) or fragment?
\item \textbf{CBDCs:} Will they coexist with crypto, or attempt to crowd it out?
\item \textbf{Institutional Adoption:} Will \$1T+ institutional capital enter crypto?
\item \textbf{DeFi vs CeFi:} Which model wins?
\item \textbf{Privacy:} Can ZK proofs enable privacy without enabling crime?
\item \textbf{Quantum Threat:} Will post-quantum cryptography be deployed in time?
\item \textbf{Mainstream UX:} Will account abstraction achieve Web2-level usability?
\end{enumerate}
\end{frame}

\begin{frame}{Realistic Scenario Details (2030)}

\begin{frame}{Final Thoughts: What Did We Learn?}
\begin{enumerate}
\item \textbf{Technology Enables, Society Decides:} Blockchain is a tool, outcomes depend on adoption
\item \textbf{Decentralization is Hard:} Tradeoffs everywhere (speed, cost, security, usability)
\item \textbf{Finance is Being Rebuilt:} From first principles, in public, with open source
\item \textbf{Speculation Funds Innovation:} Bubbles are destructive but also fund R\&D
\item \textbf{Security is Existential:} One exploit can destroy years of progress
\item \textbf{Regulation is Inevitable:} Question is whether it enables or stifles innovation
\item \textbf{User Experience:} Crypto won't achieve mainstream adoption until UX matches Web2
\item \textbf{The Future is Uncertain:} Could be revolutionary or a niche curiosity
\end{enumerate}
\end{frame}
