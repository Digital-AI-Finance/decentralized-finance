\item Understand different stablecoin mechanism types

\begin{frame}{What is a Stablecoin?}
\textbf{Definition:} A cryptocurrency designed to maintain a stable value relative to a reference asset (usually \$1 USD).

\vspace{0.3cm}
\textbf{Why Stablecoins?}
\begin{itemize}
\item \textbf{Price stability:} Avoid crypto volatility
\item \textbf{Medium of exchange:} Pay for goods/services
\item \textbf{DeFi building block:} Lending, trading, yield farming
\item \textbf{Store of value:} Park funds without exiting crypto
\item \textbf{Remittances:} Fast, cheap cross-border payments
\end{itemize}

\vspace{0.3cm}
\textbf{Total Stablecoin Market Cap:} ~\$140B (Dec 2024)

\vspace{0.3cm}
\textbf{Top Stablecoins:}
\begin{enumerate}
\item USDT (Tether): ~\$90B
\item USDC (Circle): ~\$35B
\item DAI (MakerDAO): ~\$5B
\end{enumerate}
\end{frame}

\begin{frame}{Stablecoin Trilemma}
\textbf{Challenge:} Hard to achieve all three simultaneously.

\vspace{0.3cm}
\begin{enumerate}
\item \textbf{Stability}
\begin{itemize}
\item Maintain peg to target (e.g., \$1)
\end{itemize}

\item \textbf{Decentralization}
\begin{itemize}
\item No central authority, censorship-resistant
\end{itemize}

\item \textbf{Capital Efficiency}
\begin{itemize}
\item Minimal collateral required
\end{itemize}
\end{enumerate}

\vspace{0.3cm}
\textbf{Trade-offs:}
\begin{itemize}
\item \textbf{Fiat-backed (USDC):} Stable + efficient, but centralized
\item \textbf{Crypto-backed (DAI):} Stable + decentralized, but capital inefficient
\item \textbf{Algorithmic (failed UST):} Decentralized + efficient, but unstable
\end{itemize}
\end{frame}

\begin{frame}{Type 1: Fiat-Backed Stablecoins}
\textbf{Mechanism:} 1 stablecoin backed by \$1 in bank account.

\vspace{0.3cm}
\textbf{How It Works:}
\begin{enumerate}
\item User deposits \$100 fiat to issuer
\item Issuer mints 100 stablecoin tokens
\item User can redeem 100 tokens for \$100 fiat anytime
\item Issuer holds reserves (cash, T-bills, commercial paper)
\end{enumerate}

\vspace{0.3cm}
\textbf{Examples:}
\begin{itemize}
\item \textbf{USDC (Circle):} Fully backed by cash and short-term US Treasuries, monthly attestations
\item \textbf{USDT (Tether):} Mixed reserves, less transparent, largest stablecoin
\item \textbf{BUSD (Binance):} Regulated by NYDFS (shut down 2023)
\end{itemize}

\vspace{0.3cm}
\textbf{Advantages:}
\begin{itemize}
\item Strong peg stability
\item Capital efficient (1:1 backing)
\item Easy to understand
\end{itemize}
\end{frame}

\begin{frame}{USDC: Circle's Stablecoin}
\textbf{Issuer:} Circle (US fintech company)

\vspace{0.3cm}
\textbf{Backing:}
\begin{itemize}
\item 100\% reserves in cash and short-term US Treasuries
\item Held at regulated banks
\item Monthly attestations by Grant Thornton (accounting firm)
\end{itemize}

\vspace{0.3cm}
\textbf{Minting/Burning:}
\begin{itemize}
\item Institutions can mint/redeem 1:1 with Circle
\item Retail users buy on exchanges
\end{itemize}

\vspace{0.3cm}
\textbf{Regulation:}
\begin{itemize}
\item Subject to US money transmission laws
\item Can freeze addresses (OFAC sanctions compliance)
\end{itemize}

\vspace{0.3cm}
\textbf{Market Cap:} ~\$35B (Dec 2024), deployed on Ethereum, Solana, Avalanche, Polygon, etc.
\end{frame}

\begin{frame}{USDT: Tether Controversy}
\textbf{Largest stablecoin (~\$90B) but controversial.}

\vspace{0.3cm}
\textbf{Issues:}
\begin{itemize}
\item \textbf{Opacity:} Reserves not fully transparent until recent years
\item \textbf{Commercial Paper:} Historically held risky commercial paper (reduced since 2022)
\item \textbf{Legal Issues:} Fined by CFTC (\$41M, 2021) for misleading reserve claims
\item \textbf{Banking:} No clear disclosure of reserve custodians
\end{itemize}

\vspace{0.3cm}
\textbf{Current Reserves (2024):}
\begin{itemize}
\item ~80\% US Treasuries
\item ~10\% cash
\item ~5\% corporate bonds
\item ~5\% other (Bitcoin, gold, loans)
\end{itemize}

\vspace{0.3cm}
\textbf{Why Still Dominant?}
\begin{itemize}
\item First-mover advantage (launched 2014)
\item Deepest liquidity on all exchanges
\item Network effects
\end{itemize}
\end{frame}

\begin{frame}{Fiat-Backed: Pros and Cons}
\begin{columns}[T]
\begin{column}{0.48\textwidth}
\textbf{Advantages}
\begin{itemize}
\item Strong peg (usually)
\item Capital efficient (1:1)
\item Redeemable for fiat
\item Widely accepted
\item Regulated (institutional trust)
\end{itemize}
\end{column}

\begin{column}{0.48\textwidth}
\textbf{Disadvantages}
\begin{itemize}
\item Centralized (issuer control)
\item Censorship risk (freeze funds)
\item Counterparty risk (trust issuer)
\item Opaque reserves (sometimes)
\item KYC required for redemption
\end{itemize}
\end{column}
\end{columns}

\vspace{0.3cm}
\textbf{Black Swan Risk:} Bank run if users lose confidence in reserves.

\textbf{Historical Example:} Silicon Valley Bank collapse (March 2023) briefly depegged USDC to \$0.87 (Circle had \$3.3B deposited there).
\end{frame}

\begin{frame}{Type 2: Crypto-Backed Stablecoins}
\textbf{Mechanism:} Backed by crypto collateral (overcollateralized).

\vspace{0.3cm}
\textbf{How It Works:}
\begin{enumerate}
\item User deposits \$150 of ETH as collateral
\item Protocol mints \$100 DAI stablecoin
\item Collateralization ratio: 150\% (over-backed for safety)
\item If ETH price drops, position liquidated to protect peg
\end{enumerate}

\vspace{0.3cm}
\textbf{Example: DAI (MakerDAO)}
\begin{itemize}
\item Largest decentralized stablecoin (~\$5B)
\item Backed by ETH, WBTC, USDC, and other assets
\item Governed by MKR token holders
\end{itemize}

\vspace{0.3cm}
\textbf{Advantages:}
\begin{itemize}
\item Decentralized (no central issuer)
\item Transparent (all on-chain)
\item Censorship-resistant
\end{itemize}
\end{frame}

\begin{frame}{MakerDAO: How DAI Works}
\textbf{Core Components:}

\vspace{0.3cm}
\textbf{1. Collateralized Debt Positions (CDPs) / Vaults}
\begin{itemize}
\item User deposits collateral (e.g., ETH)
\item Mints DAI up to collateralization ratio limit
\item Pays stability fee (interest) to protocol
\end{itemize}

\textbf{2. Collateralization Ratio}
\begin{itemize}
\item Minimum: 150\% for ETH (can borrow \$100 DAI with \$150 ETH)
\item Varies by collateral type (WBTC, stETH, etc.)
\end{itemize}

\textbf{3. Liquidation}
\begin{itemize}
\item If collateral value drops below threshold, vault liquidated
\item Auction mechanism sells collateral to repay DAI debt
\end{itemize}

\textbf{4. Governance (MKR Token)}
\begin{itemize}
\item Set collateral types, ratios, stability fees
\item Manage risk parameters
\end{itemize}
\end{frame}

\begin{frame}{DAI Minting Example}
\textbf{User Actions:}

\vspace{0.3cm}
\textbf{Step 1: Open Vault}
\begin{itemize}
\item Deposit 10 ETH (worth \$20,000 at \$2,000/ETH)
\end{itemize}

\textbf{Step 2: Mint DAI}
\begin{itemize}
\item Collateralization ratio: 150\%
\item Max DAI: $\frac{20{,}000}{1.5} = 13{,}333$ DAI
\item User mints 10,000 DAI (safe margin)
\end{itemize}

\textbf{Step 3: Use DAI}
\begin{itemize}
\item Spend DAI for expenses, DeFi, trading
\end{itemize}

\textbf{Step 4: Repay and Close}
\begin{itemize}
\item Repay 10,000 DAI + stability fee (e.g., 2\% annually)
\item Withdraw 10 ETH collateral
\end{itemize}

\vspace{0.3cm}
\textbf{Risk:} If ETH drops to \$1,333, vault liquidated (collateral ratio falls below 150\%).
\end{frame}

\begin{frame}{DAI Stability Mechanisms}
\textbf{How DAI maintains \$1 peg:}

\vspace{0.3cm}
\textbf{1. Arbitrage}
\begin{itemize}
\item If DAI > \$1: Mint DAI (cheap), sell for \$1+ (profit)
\item If DAI < \$1: Buy DAI (cheap), repay vault (profit)
\end{itemize}

\textbf{2. Stability Fee (Borrow Rate)}
\begin{itemize}
\item High fee: Reduces DAI supply (repayments exceed new minting)
\item Low fee: Increases DAI supply (more minting)
\end{itemize}

\textbf{3. DAI Savings Rate (DSR)}
\begin{itemize}
\item Users lock DAI to earn yield
\item Reduces circulating supply, supports peg
\end{itemize}

\textbf{4. Peg Stability Module (PSM)}
\begin{itemize}
\item Swap USDC for DAI 1:1 (no slippage)
\item Backstop for large depeg events
\end{itemize}
\end{frame}

\begin{frame}{DAI's Centralization Dilemma}
\textbf{Original Vision:} Pure crypto-backed, decentralized stablecoin.

\vspace{0.3cm}
\textbf{Current Reality (2024):}
\begin{itemize}
\item ~40\% backed by USDC (centralized stablecoin)
\item ~30\% backed by ETH/crypto
\item ~20\% backed by Real-World Assets (RWAs: T-bills, bonds)
\item ~10\% other
\end{itemize}

\vspace{0.3cm}
\textbf{Trade-off:}
\begin{itemize}
\item \textbf{USDC backing:} Improves peg stability, capital efficiency
\item \textbf{Cost:} Reduces decentralization (Circle can freeze USDC)
\end{itemize}

\vspace{0.3cm}
\textbf{Community Debate:}
\begin{itemize}
\item Should DAI maximize decentralization or stability?
\item ``Endgame'' proposal aims to reduce USDC exposure
\end{itemize}
\end{frame}

\begin{frame}{Type 3: Algorithmic Stablecoins}
\textbf{Mechanism:} Maintain peg through algorithmic supply adjustments (no collateral or minimal collateral).

\vspace{0.3cm}
\textbf{How It Works (Theoretically):}
\begin{itemize}
\item If price > \$1: Increase supply (mint new tokens, sell for profit)
\item If price < \$1: Decrease supply (buy and burn tokens, or incentivize burns)
\end{itemize}

\vspace{0.3cm}
\textbf{Examples:}
\begin{itemize}
\item \textbf{Terra UST:} Failed spectacularly (May 2022, next lecture)
\item \textbf{AMPL (Ampleforth):} Rebase token (adjusts balances daily)
\item \textbf{FRAX:} Hybrid (partially collateralized, partially algorithmic)
\end{itemize}

\vspace{0.3cm}
\textbf{Theoretical Advantage:}
\begin{itemize}
\item Capital efficient (no or minimal collateral)
\item Decentralized (no issuer)
\item Scalable
\end{itemize}

\vspace{0.3cm}
\textbf{Reality:} Most pure algorithmic stablecoins have failed (death spirals).
\end{frame}

\begin{frame}{Why Algorithmic Stablecoins Fail}
\textbf{Death Spiral Scenario:}

\vspace{0.3cm}
\begin{enumerate}
\item Stablecoin depegs below \$1 (e.g., \$0.95)
\item Protocol mints sister token to incentivize buying stablecoin
\item Sister token price crashes (dilution from minting)
\item Loss of confidence accelerates stablecoin selling
\item Stablecoin drops further (\$0.80, \$0.50, \$0.10...)
\item \textbf{Result:} Spiral to zero, total collapse
\end{enumerate}

\vspace{0.3cm}
\textbf{Root Cause:} No intrinsic backing or demand floor.

\vspace{0.3cm}
\textbf{Lesson:} Algorithmic stability relies entirely on market confidence. Once lost, irrecoverable.

\vspace{0.3cm}
\textbf{Analogy:} Like a central bank with infinite money printer but no reserves.
\end{frame}

\begin{frame}{FRAX: Hybrid Model}
\textbf{Innovation:} Partially collateralized, partially algorithmic.

\vspace{0.3cm}
\textbf{Mechanism:}
\begin{itemize}
\item Collateral Ratio (CR) adjusts dynamically (e.g., 85\% collateral, 15\% algorithmic)
\item To mint 1 FRAX: Deposit \$0.85 USDC + burn \$0.15 FXS (governance token)
\item To redeem 1 FRAX: Receive \$0.85 USDC + \$0.15 FXS
\end{itemize}

\vspace{0.3cm}
\textbf{CR Adjustment:}
\begin{itemize}
\item If FRAX > \$1: Lower CR (reduce collateral requirement)
\item If FRAX < \$1: Raise CR (increase collateral backing)
\end{itemize}

\vspace{0.3cm}
\textbf{Advantage:}
\begin{itemize}
\item More capital efficient than 100\% collateralized (DAI)
\item More stable than pure algorithmic (UST)
\end{itemize}

\vspace{0.3cm}
\textbf{Market Cap:} ~\$600M (Dec 2024), survived Terra collapse.
\end{frame}

\textbf{Observations:}

\end{itemize}
\end{frame}

\begin{frame}{Stablecoin Use Cases}
\textbf{1. Trading Pair}
\begin{itemize}
\item Most crypto traded against USDT/USDC (not USD)
\end{itemize}

\textbf{2. DeFi Lending}
\begin{itemize}
\item Earn 3-10\% APY on stablecoin deposits
\end{itemize}

\textbf{3. Payments}
\begin{itemize}
\item Remittances, merchant payments (fast, low-cost)
\end{itemize}

\textbf{4. Liquidity Provision}
\begin{itemize}
\item Low impermanent loss for stablecoin pairs (USDC/DAI)
\end{itemize}

\textbf{5. Hedging}
\begin{itemize}
\item Exit volatility without converting to fiat
\end{itemize}

\textbf{6. Yield Farming}
\begin{itemize}
\item Stable base for leveraged strategies
\end{itemize}
\end{frame}

\begin{frame}{Regulatory Landscape}
\textbf{US Regulation (Evolving):}
\begin{itemize}
\item \textbf{USDC, USDT:} May be classified as securities or payment systems
\item \textbf{Stablecoin Bill (proposed):} Require reserves, FDIC insurance, regular audits
\item \textbf{SEC View:} Some stablecoins may be securities if yield-bearing
\end{itemize}

\vspace{0.3cm}
\textbf{EU MiCA Framework (2024):}
\begin{itemize}
\item Stablecoins classified as E-Money Tokens
\item Reserve requirements (1:1 backing)
\item Issuer licensing
\item Consumer protections
\end{itemize}

\vspace{0.3cm}
\textbf{Impact:}
\begin{itemize}
\item Fiat-backed stablecoins: More regulation, higher barriers
\item Decentralized stablecoins (DAI): Regulatory gray area
\item Algorithmic stablecoins: Likely banned or heavily restricted
\end{itemize}
\end{frame}

\begin{frame}{Depeg Events}
\textbf{Historical Depegs:}

\vspace{0.3cm}
\textbf{1. USDC (March 2023)}
\begin{itemize}
\item Silicon Valley Bank collapse (\$3.3B USDC reserves)
\item USDC briefly dropped to \$0.87
\item Recovered within days (FDIC backstop, Circle transparency)
\end{itemize}

\textbf{2. DAI (Black Thursday, March 2020)}
\begin{itemize}
\item ETH crashed 50\% in hours
\item Liquidations overwhelmed system
\item DAI briefly spiked to \$1.10 (high demand)
\end{itemize}

\textbf{3. USDT (Multiple Times)}
\begin{itemize}
\item Briefly depegged to \$0.95-\$0.98 during FUD
\item Always recovered (deep liquidity)
\end{itemize}

\vspace{0.3cm}
\textbf{Lesson:} Even well-backed stablecoins can temporarily depeg during extreme stress.
\end{frame}

\begin{frame}{Reserve Transparency}
\textbf{Audit Levels (Best to Worst):}

\vspace{0.3cm}
\begin{enumerate}
\item \textbf{Full Audit}
\begin{itemize}
\item Independent auditor verifies reserves
\item Rare (expensive, time-consuming)
\end{itemize}

\item \textbf{Attestation}
\begin{itemize}
\item Accountant confirms reserve snapshot
\item More common (USDC, USDT)
\end{itemize}

\item \textbf{Self-Reporting}
\begin{itemize}
\item Issuer publishes own data
\item Least trustworthy
\end{itemize}

\item \textbf{On-Chain (Crypto-Backed)}
\begin{itemize}
\item Reserves visible on blockchain (DAI)
\item Most transparent
\end{itemize}
\end{enumerate}

\vspace{0.3cm}
\textbf{Best Practice:} Monthly attestations minimum, real-time on-chain tracking preferred.
\end{frame}

\begin{frame}{Summary}
\textbf{Key Takeaways:}
\begin{itemize}
\item Stablecoins face trilemma: stability, decentralization, capital efficiency
\item Fiat-backed (USDC, USDT) dominate market cap (~90\%) but are centralized
\item Crypto-backed (DAI) is decentralized but capital inefficient (150\% collateralization)
\item Algorithmic stablecoins (UST) have repeatedly failed via death spirals
\item Hybrid models (FRAX) balance collateral and algorithm
\item Depegs can occur even with strong backing (SVB crisis affected USDC)
\item Regulatory focus intensifying (reserve requirements, licensing)
\end{itemize}
