\begin{frame}{Terra Ecosystem Overview}
\textbf{Key Components:}

\vspace{0.3cm}
\begin{itemize}
\item \textbf{UST (TerraUSD):} Algorithmic stablecoin pegged to \$1
\item \textbf{LUNA:} Native token of Terra blockchain, used to stabilize UST
\item \textbf{Anchor Protocol:} Lending platform offering ~20\% APY on UST deposits
\item \textbf{Terra Blockchain:} Layer 1 built on Cosmos SDK
\end{itemize}

\vspace{0.3cm}
\textbf{Creator:} Do Kwon (Terraform Labs)

\vspace{0.3cm}
\textbf{Peak Market Cap (April 2022):}
\begin{itemize}
\item UST: \$18 billion
\item LUNA: \$40 billion
\item Combined: \$58 billion ecosystem
\end{itemize}

\vspace{0.3cm}
\textbf{Collapse (May 2022):} Both UST and LUNA went to near-zero in 72 hours.
\end{frame}

\begin{frame}{The UST-LUNA Mechanism}
\textbf{Algorithmic Stability:} UST maintains \$1 peg through arbitrage with LUNA.

\vspace{0.3cm}
\textbf{Mechanism:}
\begin{enumerate}
\item \textbf{Mint UST:} Burn \$1 of LUNA, mint 1 UST
\item \textbf{Burn UST:} Burn 1 UST, mint \$1 of LUNA
\end{enumerate}

\vspace{0.3cm}
\textbf{Arbitrage Examples:}

\textbf{If UST > \$1 (e.g., \$1.05):}
\begin{itemize}
\item Burn \$1 LUNA, mint 1 UST
\item Sell 1 UST for \$1.05
\item Profit: \$0.05 per UST
\item Effect: UST supply increases, price falls back to \$1
\end{itemize}

\textbf{If UST < \$1 (e.g., \$0.95):}
\begin{itemize}
\item Buy 1 UST for \$0.95
\item Burn 1 UST, mint \$1 LUNA
\item Profit: \$0.05 per UST
\item Effect: UST supply decreases, price rises back to \$1
\end{itemize}
\end{frame}

\begin{frame}{The Fatal Flaw}
\textbf{Problem:} Mechanism only works if LUNA has value.

\vspace{0.3cm}
\textbf{Death Spiral Logic:}
\begin{enumerate}
\item UST depegs below \$1 (e.g., \$0.90)
\item Arbitrageurs burn UST, mint LUNA
\item Massive LUNA supply increase (dilution)
\item LUNA price crashes
\item Confidence in UST collapses (if LUNA worthless, UST unbacked)
\item More UST selling, deeper depeg
\item Cycle repeats, accelerating downward
\end{enumerate}

\vspace{0.3cm}
\textbf{Analogy:} Like a central bank defending currency by printing more currency (hyperinflation).

\vspace{0.3cm}
\textbf{Critical Insight:} Pure algorithmic stablecoins have no demand floor. Once confidence breaks, irrecoverable.
\end{frame}

\begin{frame}{Anchor Protocol: The 20\% APY Trap}
\textbf{Promise:} Earn ~20\% APY by depositing UST.

\vspace{0.3cm}
\textbf{How It Worked:}
\begin{itemize}
\item Users deposit UST into Anchor
\item Anchor lends UST to borrowers (collateralized by staked LUNA, ETH)
\item Borrowers pay interest (~10-15\%)
\item \textbf{Gap:} Anchor subsidized difference to reach 20\% (unsustainable)
\end{itemize}

\vspace{0.3cm}
\textbf{Subsidy Source:}
\begin{itemize}
\item Terraform Labs treasury (LUNA token sales)
\item Yield Reserve (depleting over time)
\end{itemize}

\vspace{0.3cm}
\textbf{Red Flag:}
\begin{itemize}
\item Yield reserve burning ~\$50-100M per month
\item Projected to run out by mid-2022
\item Community votes to reduce APY rejected
\end{itemize}
\end{frame}

\begin{frame}{Why 20\% APY Was Unsustainable}
\textbf{Math Doesn't Add Up:}

\vspace{0.3cm}
\textbf{Anchor's Revenue:}
\begin{itemize}
\item Borrow interest: ~12\% APY on \$4B borrowed = \$480M/year
\item Other income: Minimal
\end{itemize}

\textbf{Anchor's Expenses:}
\begin{itemize}
\item Deposit interest: 20\% APY on \$14B deposited = \$2.8B/year
\end{itemize}

\textbf{Annual Deficit:}
\[
\$2.8B - \$480M = \$2.32B \text{ shortfall}
\]

\vspace{0.3cm}
\textbf{Conclusion:} Anchor was a Ponzi-like scheme. New deposits paid old depositors until subsidy ran out.

\vspace{0.3cm}
\textbf{Warning Sign:} No sustainable business model can pay 20\% in a low-rate environment.
\end{frame}

\begin{frame}{Timeline: The Collapse (May 7-12, 2022)}
\textbf{May 7-8: Initial Depeg}
\begin{itemize}
\item Large UST withdrawals from Anchor (\$2B+ in 48 hours)
\item Curve 3pool (UST/USDC/USDT) imbalanced
\item UST briefly drops to \$0.98
\end{itemize}

\textbf{May 9: Panic Begins}
\begin{itemize}
\item UST falls to \$0.92
\item \$1B UST burned, minting billions of LUNA
\item LUNA price drops 30\% (\$80 $\to$ \$60)
\end{itemize}

\textbf{May 10: Death Spiral Accelerates}
\begin{itemize}
\item UST at \$0.60
\item LUNA supply balloons (100M $\to$ 1B tokens in hours)
\item LUNA price collapses (\$60 $\to$ \$1)
\end{itemize}

\textbf{May 11-12: Total Collapse}
\begin{itemize}
\item UST at \$0.10, then \$0.01
\item LUNA at \$0.0001 (6.5 trillion tokens in circulation)
\item Terra blockchain halted
\end{itemize}
\end{frame}

\begin{frame}{The Numbers: Destruction of Value}
\textbf{Pre-Collapse (Early May 2022):}
\begin{itemize}
\item UST market cap: \$18B
\item LUNA market cap: \$40B
\item Anchor deposits: \$14B
\item Total ecosystem: ~\$60B
\end{itemize}

\vspace{0.3cm}
\textbf{Post-Collapse (Mid-May 2022):}
\begin{itemize}
\item UST value: ~\$0 (99\%+ loss)
\item LUNA value: ~\$0 (99.9\%+ loss)
\item Estimated losses: \$40B+ (investors, users, protocols)
\end{itemize}

\vspace{0.3cm}
\textbf{Casualties:}
\begin{itemize}
\item Retail investors: Life savings wiped out
\item VCs: Jump Crypto, Three Arrows Capital (3AC later collapsed)
\item DeFi protocols: Many held UST reserves, suffered losses
\end{itemize}
\end{frame}

\begin{frame}{Bitcoin Reserve: Failed Rescue}
\textbf{Luna Foundation Guard (LFG):}
\begin{itemize}
\item Reserve fund created to defend UST peg
\item Raised \$3B+ in Bitcoin, AVAX, other assets
\end{itemize}

\vspace{0.3cm}
\textbf{Plan:}
\begin{itemize}
\item If UST depegs, sell BTC to buy UST (support price)
\end{itemize}

\vspace{0.3cm}
\textbf{Reality (May 2022):}
\begin{itemize}
\item LFG deployed ~\$1.5B BTC to buy UST
\item Ineffective: Death spiral too fast
\item Lost most of the BTC (sold near bottom)
\item Remaining reserves ($\sim$\$100M) insufficient
\end{itemize}

\vspace{0.3cm}
\textbf{Lesson:} Finite reserves cannot defend algorithmic peg under mass panic. Only collateralization works.
\end{frame}

\begin{frame}{Three Arrows Capital (3AC) Contagion}
\textbf{Connection:}
\begin{itemize}
\item 3AC was major LUNA investor
\item Held billions in LUNA tokens
\end{itemize}

\vspace{0.3cm}
\textbf{Cascade:}
\begin{enumerate}
\item LUNA collapses (May 2022)
\item 3AC's collateral (LUNA) becomes worthless
\item 3AC fails to meet margin calls on other loans
\item Lenders (BlockFi, Celsius, Voyager) take massive losses
\item 3AC declares bankruptcy (July 2022)
\item CeFi contagion spreads: Celsius, Voyager, BlockFi all fail
\end{enumerate}

\vspace{0.3cm}
\textbf{Impact:} Terra collapse triggered \$200B+ broader crypto crash, multiple bankruptcies.

\vspace{0.3cm}
\textbf{Lesson:} Interconnected leverage amplifies failures. Terra was the first domino.
\end{frame}

\begin{frame}{Do Kwon's Hubris}
\textbf{Quotes (Pre-Collapse):}

\vspace{0.3cm}
\begin{itemize}
\item ``I don't debate the poor.'' (Twitter, responding to critics)
\item ``95\% of stablecoins will fail, UST will win.''
\item ``Algorithmic stablecoins are the future of money.''
\end{itemize}

\vspace{0.3cm}
\textbf{Behavior:}
\begin{itemize}
\item Dismissed warnings from economists, DeFi experts
\item Aggressive marketing (``Lunatics'' community)
\item Promised unrealistic yields (20\% Anchor)
\end{itemize}

\vspace{0.3cm}
\textbf{Legal Aftermath:}
\begin{itemize}
\item SEC charged Do Kwon with fraud (2023)
\item Interpol arrest warrant issued
\item Arrested in Montenegro (March 2023)
\item Facing extradition to US or South Korea
\end{itemize}

\vspace{0.3cm}
\textbf{Charges:} Securities fraud, misleading investors, market manipulation.
\end{frame}

\begin{frame}{Warning Signs (In Hindsight)}
\textbf{Red Flags That Were Ignored:}

\vspace{0.3cm}
\begin{enumerate}
\item \textbf{Unsustainable Yield}
\begin{itemize}
\item 20\% APY with no underlying revenue
\end{itemize}

\item \textbf{Algorithmic Design Flaw}
\begin{itemize}
\item No collateral backing, death spiral risk
\end{itemize}

\item \textbf{Concentration Risk}
\begin{itemize}
\item \$14B of \$18B UST locked in Anchor (78\%)
\end{itemize}

\item \textbf{Hubris and Dismissiveness}
\begin{itemize}
\item Founder ignored criticism, attacked skeptics
\end{itemize}

\item \textbf{Rapid Growth}
\begin{itemize}
\item UST grew from \$2B to \$18B in 6 months (too fast)
\end{itemize}

\item \textbf{Lack of Stress Testing}
\begin{itemize}
\item No proven resilience during market downturns
\end{itemize}
\end{enumerate}
\end{frame}

\begin{frame}{Lessons Learned}
\textbf{For Investors:}
\begin{itemize}
\item Beware of unsustainable yields (>10\% on stablecoins)
\item Algorithmic stablecoins are experimental and risky
\item Diversify across stablecoin types (don't go all-in on one)
\item Hubris and dismissiveness are red flags
\end{itemize}

\textbf{For Protocols:}
\begin{itemize}
\item Overcollateralization is necessary for stability
\item Stress test designs under extreme scenarios
\item Don't rely on perpetual growth for sustainability
\item Transparency and humility attract trust
\end{itemize}

\textbf{For Regulators:}
\begin{itemize}
\item Algorithmic stablecoins pose systemic risk
\item Consider reserve requirements for stablecoins
\item Protect retail investors from Ponzi-like yields
\end{itemize}
\end{frame}

\begin{frame}{Survivor: DAI vs. UST}
\textbf{Why did DAI survive while UST collapsed?}

\vspace{0.3cm}
\begin{columns}[T]
\begin{column}{0.48\textwidth}
\textbf{UST (Failed)}
\begin{itemize}
\item No collateral
\item Pure algorithm
\item Relied on LUNA value
\item Death spiral possible
\item Unsustainable yield (Anchor)
\end{itemize}
\end{column}

\begin{column}{0.48\textwidth}
\textbf{DAI (Survived)}
\begin{itemize}
\item 150\%+ overcollateralization
\item Backed by ETH, USDC, etc.
\item Independent of one token
\item Liquidations prevent death spiral
\item No unrealistic yield promises
\end{itemize}
\end{column}
\end{columns}

\vspace{0.3cm}
\textbf{Key Difference:} Collateral provides demand floor. If DAI < \$1, arbitrageurs buy DAI to repay vaults profitably.

\vspace{0.3cm}
\textbf{Conclusion:} Overcollateralization is inefficient but necessary for stability.
\end{frame}

\begin{frame}{Terra 2.0: Failed Reboot}
\textbf{Post-Collapse Plan (May 2022):}
\begin{itemize}
\item Abandon UST (stablecoin)
\item Relaunch Terra blockchain as ``Terra 2.0''
\item New LUNA token (non-algorithmic)
\item Airdrop to UST/LUNA holders (compensation attempt)
\end{itemize}

\vspace{0.3cm}
\textbf{Result:}
\begin{itemize}
\item Terra 2.0 launched with minimal adoption
\item New LUNA price: \$6 $\to$ \$0.50 (within months)
\item Community fractured, trust destroyed
\item Ecosystem projects migrated to other chains
\end{itemize}

\vspace{0.3cm}
\textbf{Lesson:} Trust, once broken, is nearly impossible to rebuild. Rebranding doesn't erase history.
\end{frame}

\begin{frame}{Regulatory Fallout}
\textbf{US Response:}
\begin{itemize}
\item SEC charged Do Kwon with fraud
\item Hearings on stablecoin regulation
\item Proposed legislation requiring reserve backing
\end{itemize}

\textbf{Global Response:}
\begin{itemize}
\item South Korea: Investigation into Terraform Labs
\item Singapore: Tightened crypto regulations
\item EU MiCA: Included stablecoin reserve requirements
\end{itemize}

\vspace{0.3cm}
\textbf{Industry Impact:}
\begin{itemize}
\item Algorithmic stablecoins largely abandoned
\item FRAX increased collateral ratio to 100\%
\item Focus shifted to fiat-backed and overcollateralized models
\end{itemize}

\vspace{0.3cm}
\textbf{Long-term:} Terra collapse accelerated stablecoin regulation worldwide.
\end{frame}

\begin{frame}{Psychological Factors}
\textbf{Why Did People Believe?}

\vspace{0.3cm}
\begin{enumerate}
\item \textbf{Greed}
\begin{itemize}
\item 20\% APY too attractive to pass up
\end{itemize}

\item \textbf{FOMO}
\begin{itemize}
\item Fear of missing out on ``next big thing''
\end{itemize}

\item \textbf{Herd Mentality}
\begin{itemize}
\item Everyone else was buying (social proof)
\end{itemize}

\item \textbf{Confirmation Bias}
\begin{itemize}
\item Ignored critics, only listened to bulls
\end{itemize}

\item \textbf{Complexity}
\begin{itemize}
\item Most users didn't understand the mechanism
\end{itemize}

\item \textbf{Charismatic Leader}
\begin{itemize}
\item Do Kwon's confidence was persuasive
\end{itemize}
\end{enumerate}

\vspace{0.3cm}
\textbf{Lesson:} Always question high yields. If it sounds too good to be true, it probably is.
\end{frame}

\begin{frame}{Comparison: Terra vs. Traditional Bank Run}
\begin{columns}[T]
\begin{column}{0.48\textwidth}
\textbf{Traditional Bank Run}
\begin{itemize}
\item Fractional reserves (10\% capital)
\item FDIC insurance (\$250k/account)
\item Central bank lender of last resort
\item Slow (days to withdraw)
\item Government can freeze
\end{itemize}
\end{column}

\begin{column}{0.48\textwidth}
\textbf{Terra Death Spiral}
\begin{itemize}
\item Zero reserves (algorithmic)
\item No insurance
\item No lender of last resort
\item Instant (blockchain speed)
\item Unstoppable (permissionless)
\end{itemize}
\end{column}
\end{columns}

\vspace{0.3cm}
\textbf{Key Difference:} Traditional finance has circuit breakers. DeFi operates 24/7 with no pause button.

\vspace{0.3cm}
\textbf{Implication:} DeFi failures can be faster and more catastrophic, but also more transparent.
\end{frame}

\begin{frame}{Summary}
\textbf{Key Takeaways:}
\begin{itemize}
\item Terra/Luna was \$60B algorithmic stablecoin experiment that collapsed in May 2022
\item UST-LUNA mechanism relied on arbitrage but had no collateral backing
\item Anchor's 20\% APY was unsustainable Ponzi-like scheme
\item Death spiral: UST depeg $\to$ LUNA hyperinflation $\to$ total collapse
\item \$40B+ value destroyed, triggering broader crypto contagion (3AC, Celsius)
\item Lessons: Beware unsustainable yields, algorithmic stablecoins need collateral
\item Regulatory response: Accelerated stablecoin regulation globally
\end{itemize}
